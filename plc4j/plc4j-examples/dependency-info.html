Dependency Information

Apache Maven

<dependency>
  <groupId>org.apache.plc4x.examples</groupId>
  <artifactId>plc4j-examples</artifactId>
  <version>0.6.0-SNAPSHOT</version>
  <type>pom</type>
</dependency>

Apache Buildr

'org.apache.plc4x.examples:plc4j-examples:pom:0.6.0-SNAPSHOT'

Apache Ivy

<dependency org="org.apache.plc4x.examples" name="plc4j-examples" rev="0.6.0-SNAPSHOT">
  <artifact name="plc4j-examples" type="pom" />
</dependency>

Groovy Grape

@Grapes(
@Grab(group='org.apache.plc4x.examples', module='plc4j-examples', version='0.6.0-SNAPSHOT')
)

Gradle/Grails

compile 'org.apache.plc4x.examples:plc4j-examples:0.6.0-SNAPSHOT'

Scala SBT

libraryDependencies += "org.apache.plc4x.examples" % "plc4j-examples" % "0.6.0-SNAPSHOT"

Leiningen

[org.apache.plc4x.examples/plc4j-examples "0.6.0-SNAPSHOT"]
