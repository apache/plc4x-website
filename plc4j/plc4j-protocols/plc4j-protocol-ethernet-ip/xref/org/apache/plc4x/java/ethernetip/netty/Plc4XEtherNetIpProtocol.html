84       * If the IsoTP protocol is used on top of the ISO on TCP protocol, then as soon as the pipeline receives the
85       * request to connect, an IsoTP connection request TPDU must be sent in order to initialize the connection.

87       * @param ctx the current protocol layers context
88       * @param evt the event
89       * @throws Exception throws an exception if something goes wrong internally
90       */
91      @Override
92      public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
93          // If the connection has just been established, start setting up the connection
94          // by sending a connection request to the plc.
95          if (evt instanceof ConnectEvent) {
96              LOGGER.debug("EtherNet/IP Protocol Sending Connection Request");
97  
98              EnipPacket packet = new EnipPacket(CommandCode.RegisterSession, 0, EnipStatus.EIP_SUCCESS,
99                  messageId.getAndIncrement(), new RegisterSession());
100 
101             ctx.channel().writeAndFlush(packet);
102         } else {
103             super.userEventTriggered(ctx, evt);
104         }
105     }

176     private void encodeReadRequest(PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> msg, List<Object> out) {
177         if (!supportsCipEncapsulation) {
178             LOGGER.warn("CIP Encapsulation not supported by remote, payload encapsulation must be handled by target and originator");
179         }
180 
181         InternalPlcReadRequest request = (InternalPlcReadRequest) msg.getRequest();
182         for (String fieldName : request.getFieldNames()) {
183             PlcField field = request.getField(fieldName);
184 
185             // CIP Part
186             EtherNetIpField../../org/apache/plc4x/java/ethernetip/model/EtherNetIpField.html#EtherNetIpField">EtherNetIpField enipField = (EtherNetIpField) field;
187             EPath.PaddedEPath path = new EPath.PaddedEPath(new LogicalSegment.ClassId(enipField.getObjectNumber()),
188                 new LogicalSegment.InstanceId(enipField.getInstanceNumber()),
189                 new LogicalSegment.AttributeId(enipField.getAttributeNumber()));
190             GetAttributeSingleService service = new GetAttributeSingleService(path);
191 
192             // ENIP Part
193             EnipPacket packet = new EnipPacket(CommandCode.SendRRData, sessionHandle, EnipStatus.EIP_SUCCESS,
194                 messageId.getAndIncrement(), new SendRRData(new CpfPacket(
195                 new NullAddressItem(),
196                 new UnconnectedDataItemRequest(service::encodeRequest)
197             )));
198 
199             requestsMap.put(packet.getSenderContext(), msg);
200 
201             out.add(packet);
202         }
203     }

209     @SuppressWarnings("unchecked")
210     @Override
211     protected void decode(ChannelHandlerContext ctx, EnipPacket msg, List<Object> out) {
212         LOGGER.trace("(-->IN): {}, {}, {}", ctx, msg, out);
213         LOGGER.debug("{}: session handle: {}, sender context: {}, EtherNetIPPacket:{}", msg, msg.getSessionHandle(), msg.getSenderContext(), msg);
214 
215         EnipPacket packet = null;
216         switch (msg.getCommandCode()) {
217             case RegisterSession:
218                 handleRegisterSession(ctx, msg);
219 
220                 // Now try getting some detailed information about the remote.
221                 packet = new EnipPacket(CommandCode.ListIdentity, sessionHandle, EnipStatus.EIP_SUCCESS,
222                     messageId.getAndIncrement(), new ListIdentity());
223                 break;
224 
225             case UnRegisterSession:
226                 handleUnregisterSession(ctx, msg);
227 
228                 // Spec: The receiver shall initiate a close of the underlying
229                 // TCP/IP connection when it receives this command.
230                 ctx.channel().disconnect();
231                 break;
232 
233             case ListIdentity:
234                 handleListIdentity(ctx, msg);
235 
236                 // Now try listing the services the remote has to offer.
237                 packet = new EnipPacket(CommandCode.ListServices, sessionHandle, EnipStatus.EIP_SUCCESS,
238                     messageId.getAndIncrement(), new ListServices());
239                 break;
240 
241             case ListInterfaces:
242                 handleListInterfaces(ctx, msg);
243 
244                 // Here we're done connecting.
245                 ctx.channel().pipeline().fireUserEventTriggered(new ConnectedEvent());
246                 break;
247 
248             case ListServices:
249                 handleListServices(ctx, msg);
250 
251                 // Now try listing the interfaces the remote has to offer.
252                 packet = new EnipPacket(CommandCode.ListInterfaces, sessionHandle, EnipStatus.EIP_SUCCESS,
253                     messageId.getAndIncrement(), new ListInterfaces());
254                 break;
255 
256             case Nop:
257                 handleNop(ctx, msg);
258                 break;
259 
260             case SendRRData:
261                 handleSendRRDataResponse(ctx, msg);
262                 break;
263 
264             case SendUnitData:
265                 // This might be where the connected data is sent (eventually publish/subscribe communication)
266                 break;
267         }
268 
269         if (packet != null) {
270             ctx.channel().writeAndFlush(packet);
271         }
272     }

274     /**
275      * In order to do explicit connected messaging, the client has to register a session with the server.
276      * In case of a successful session registration the response will contain the sessionHandle, which is
277      * required to be used in all subsequent connected interactions.
278      *
279      * @param ctx the {@link ChannelHandlerContext} instance.
280      * @param msg the packet received from the server.
281      */
282     private void handleRegisterSession(ChannelHandlerContext ctx, EnipPacket msg) {
283         if (msg.getStatus() == EnipStatus.EIP_SUCCESS) {
284             sessionHandle = msg.getSessionHandle();
285 
286             LOGGER.info("EtherNet/IP session registered session-handle {}", sessionHandle);
287         } else {
288             ctx.channel().pipeline().fireExceptionCaught(new PlcProtocolException("Got a non-success response."));
289         }
290     }

292     /**
293      * As connected operations allocate resources on the server and the client, when receiving a
294      * {@link UnRegisterSession} message (request or response) the locally allocated resources have
295      * to be released again. As the correct response to a UnRegisterSession is the closing of the
296      * connection by the receiving side, this incoming command must be a request sent from the
297      * server.
298      *
299      * @param ctx the {@link ChannelHandlerContext} instance.
300      * @param msg the packet received from the server.
301      */
302     private void handleUnregisterSession(ChannelHandlerContext ctx, EnipPacket msg) {
303         if (msg.getStatus() == EnipStatus.EIP_SUCCESS) {
304             // Reset all internal variables.
305             identityItem = null;
306             supportsCipEncapsulation = false;
307             supportsClass0Or1UdpConnections = false;
308             nonCipInterfaces = null;
309             fieldConnectionMap = null;
310         } else {
311             ctx.channel().pipeline().fireExceptionCaught(new PlcProtocolException("Got a non-success response."));
312         }
313     }

315     /**
316      * The response to a {@link ListIdentity} command contains a lot of information about the
317      * remote counterpart. In this case we just save this information for further usage.
318      *
319      * @param ctx the {@link ChannelHandlerContext} instance.
320      * @param msg the packet received from the server.
321      */
322     private void handleListIdentity(ChannelHandlerContext ctx, EnipPacket msg) {
323         if (msg.getStatus() == EnipStatus.EIP_SUCCESS) {
324             ListIdentity listIdentityResponse = (ListIdentity) msg.getCommand();
325             if (listIdentityResponse != null) {
326                 identityItem = listIdentityResponse.getIdentity().orElse(null);
327                 if (identityItem != null) {
328                     LOGGER.info("Connected to: \n - product name: {} \n - serial number: {} ",
329                         identityItem.getProductName().trim(), identityItem.getSerialNumber());
330                 }
331             } else {
332                 identityItem = null;
333             }
334         } else {
335             ctx.channel().pipeline().fireExceptionCaught(new PlcProtocolException("Got a non-success response."));
336         }
337     }

339     /**
340      * Some times EtherNet/IP devices support other devices than the default one.
341      * As we are required to ev eventually reference these interfaces, build a map
342      * of all the devices the remote supports. This way we can check the validity
343      * before actually sending a request.
344      *
345      * @param ctx the {@link ChannelHandlerContext} instance.
346      * @param msg the packet received from the server.
347      */
348     private void handleListInterfaces(ChannelHandlerContext ctx, EnipPacket msg) {
349         if (msg.getStatus() == EnipStatus.EIP_SUCCESS) {
350             ListInterfaces listInterfaces = (ListInterfaces) msg.getCommand();
351             if (listInterfaces != null) {
352                 // If the device supports non-CIP interfaces, this array is not empty.
353                 // In this case build a map so we can access the information when sending
354                 // data in RR-Requests (Request-Response).
355                 if (listInterfaces.getInterfaces().length > 0) {
356                     nonCipInterfaces = new HashMap<>();
357                     for (ListInterfaces.InterfaceInformation interfaceInformation : listInterfaces.getInterfaces()) {
358                         String interfaceName = new String(
359                             interfaceInformation.getData(), Charset.forName("US-ASCII")).trim();
360                         nonCipInterfaces.put(interfaceName, interfaceInformation.hashCode());
361                     }
362                 } else {
363                     nonCipInterfaces = null;
364                 }
365             }
366         } else {
367             ctx.channel().pipeline().fireExceptionCaught(new PlcProtocolException("Got a non-success response."));
368         }
369     }

371     /**
372      * Each EtherNet/IP device can support one or more so-called `services`. At least the `Communications`
373      * service is required to be supported by every EtherNet/IP compliant device. This is used for default
374      * IO operations. Usually vendors support custom services which are adjusted to their particular needs,
375      * which might be able to provide better performance than the default. In this case we are ignoring all
376      * these as supporting these would require custom adapters on the PLC4X side. However we do inspect the
377      * capabilities of the `Communications` service to check if encapsulation of CIP data is supported and
378      * if we are able to do connected implicit communication via a parallel UDP channel.
379      *
380      * @param ctx the {@link ChannelHandlerContext} instance.
381      * @param msg the packet received from the server.
382      */
383     private void handleListServices(ChannelHandlerContext ctx, EnipPacket msg) {
384         if (msg.getStatus() == EnipStatus.EIP_SUCCESS) {
385             ListServices listServices = (ListServices) msg.getCommand();
386             if (listServices != null) {
387                 for (ListServices.ServiceInformation service : listServices.getServices()) {
388                     // Check if the type code matches the communications service and if bit 5 of the
389                     // capability flags is set.
390                     if (service.getTypeCode() == SERVICE_COMMUNICATIONS_TYPE_CODE) {
391                         supportsCipEncapsulation = (service.getCapabilityFlags() & 32) != 0;
392                         supportsClass0Or1UdpConnections = (service.getCapabilityFlags() & 256) != 0;
393                     }
394                 }
395             } else {
396                 supportsCipEncapsulation = false;
397                 supportsClass0Or1UdpConnections = false;
398             }
399         } else {
400             ctx.channel().pipeline().fireExceptionCaught(new PlcProtocolException("Got a non-success response."));
401         }
402     }

422     /**
423      * As RR Data is Request Response data and the server will not issue a request to
424      * the client, we can be pretty sure this is a response to a previously issued request.
425      * This contains the actual payload for our requests.
426      *
427      * @param ctx the {@link ChannelHandlerContext} instance.
428      * @param msg the packet received from the server.
429      */
430     private void handleSendRRDataResponse(ChannelHandlerContext ctx, EnipPacket msg) {
431         // This is where the typical request/response stuff is handled.
432         long senderContext = msg.getSenderContext();
433         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> plcRequestContainer = requestsMap.get(senderContext);
434         if (plcRequestContainer == null) {
435             ctx.channel().pipeline().fireExceptionCaught(
436                 new PlcProtocolException("Unrelated payload received for message " + msg));
437             return;
438         }
439 
440         if (!(plcRequestContainer.getRequest() instanceof PlcReadRequest)) {
441             ctx.fireExceptionCaught(new PlcProtocolException("Expecting a PlcReadRequest here."));
442             return;
443         }
444         InternalPlcReadRequest request = (InternalPlcReadRequest) plcRequestContainer.getRequest();
445         PlcResponseCode responseCode;
446         if (msg.getStatus() != EnipStatus.EIP_SUCCESS) {
447             responseCode = PlcResponseCode.NOT_FOUND;
448         } else {
449             responseCode = PlcResponseCode.OK;
450         }
451 
452         SendRRData sendRRDataCommand = (SendRRData) msg.getCommand();
453         if (sendRRDataCommand == null) {
454             ctx.fireExceptionCaught(new PlcProtocolException("Expecting a SendRRData command here."));
455             return;
456         }
457         CpfItem[] items = sendRRDataCommand.getPacket().getItems();
458         if (items.length != 2) {
459             ctx.fireExceptionCaught(new PlcProtocolException("Expecting 2 items here."));
460             return;
461         }
462         CpfItem payload = items[1];
463         if (!(payload instanceof UnconnectedDataItemResponse)) {
464             ctx.fireExceptionCaught(new PlcProtocolException("Item[1] should be of type UnconnectedDataItemResponse"));
465             return;
466         }
467         UnconnectedDataItemResponse enipResponse = (UnconnectedDataItemResponse) payload;
468         ByteBuf data = enipResponse.getData();
469         if (data.readableBytes() > 0) {
470             Map<String, Pair<PlcResponseCode, BaseDefaultFieldItem>> values = new HashMap<>();
471             // TODO: This is not quite correct as this will probalby not work when requesting more than one item.
472             for (String fieldName : request.getFieldNames()) {
473                 MessageRouterResponse cipResponse = MessageRouterResponse.decode(data);
474                 short value;
475                 // TODO: This is not quite correct as we assume everything is an integer.
476                 if (cipResponse.getData().readableBytes() >= 2) {
477                     value = cipResponse.getData().readShort();
478                 } else {
479                     value = -1;
480                 }
481                 DefaultLongFieldItem fieldItem = new DefaultLongFieldItem((long) value);
482                 values.put(fieldName, new ImmutablePair<>(responseCode, fieldItem));
483             }
484             InternalPlcReadResponse response = new DefaultPlcReadResponse(request, values);
485             plcRequestContainer.getResponseFuture().complete(response);
486         }
487     }
