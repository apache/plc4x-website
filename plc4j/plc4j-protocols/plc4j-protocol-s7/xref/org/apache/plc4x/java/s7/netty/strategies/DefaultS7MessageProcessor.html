45  /**
46   * While a SetupCommunication message is no problem, when reading or writing data,
47   * situations could arise that have to be handled. The following situations have to
48   * be handled:
49   * - The number of request items is so big, that the resulting PDU would exceed the
50   *   agreed upon PDU size: The request has to be split up into multiple requests.
51   * - If large blocks of data are requested by request items the result of a request
52   *   could exceed the PDU size: The requests has to be split up into multiple requests
53   *   where each requests response doesn't exceed the PDU size.
54   *
55   * The following optimizations should be implemented:
56   * - If blocks are read which are in near proximity to each other it could be better
57   *   to replace multiple requests by one that includes multiple blocks.
58   * - Rearranging the order of request items could reduce the number of needed PDUs.
59   */
60  public class DefaultS7MessageProcessor implements S7MessageProcessor {
61  
62      private AtomicInteger tpduRefGen;
63  
64      public DefaultS7MessageProcessor() {
65          this.tpduRefGen = new AtomicInteger(1);
66      }
67  
68      @Override
69      public Collection<S7RequestMessage> processRequest(S7RequestMessage request, int pduSize) throws PlcException {
70          // The following considerations have to be taken into account:
71          // - The size of all parameters and payloads of a message cannot exceed the negotiated PDU size
72          // - When reading data, the size of the returned data cannot exceed the negotiated PDU size
73          //
74          // Examples:
75          // - Size of the request exceeds the maximum
76          //  When having a negotiated max PDU size of 256, the maximum size of individual addresses can be at most 18
77          //  If more are sent, the S7 will respond with a frame error.
78          // - Size of the response exceeds the maximum
79          //  When reading two Strings of each 200 bytes length, the size of the request is ok, however the PLC would
80          //  have to send back 400 bytes of String data, which would exceed the PDU size. In this case the first String
81          //  is correctly returned, but for the second item the PLC will return a code of 0x03 = Access Denied
82          // - A S7 device doesn't seem to accept more than one write item. So if we are doing write operations, we
83          //  have to split that up into one message per written item. This also seems to affect arrays. So if
84          //  an array of values is written, we have to also split up that array into single writes.
85  
86          Optional<VarParameter> varParameterOptional = request.getParameter(VarParameter.class);
87          if (varParameterOptional.isPresent()) {
88              VarParameter varParameter = varParameterOptional.get();
89  
90              // If this is a read operation, try to get as many items in as possible.
91              if(varParameter.getType() == ParameterType.READ_VAR) {
92                  return processReadVarParameter(request, varParameter, pduSize).getRequestMessages();
93              }
94  
95              // If this is a write operation, split up every array item into single value items
96              // and every item into a separate message.
97              else if(varParameter.getType() == ParameterType.WRITE_VAR) {
98                  return processWriteVarParameter(request, varParameter).getRequestMessages();
99              }
100         }
101 
102         return Collections.singletonList(request);
103     }

105     private S7CompositeRequestMessage processReadVarParameter(S7RequestMessage request, VarParameter varParameter, int pduSize) {
106         // Create a new composite request message.
107         S7CompositeRequestMessage compositeRequestMessage = new S7CompositeRequestMessage(request);
108 
109         // Create a var parameter without any items (yet).
110         VarParameter subVarParameter = new VarParameter(varParameter.getType(), new LinkedList<>());
111 
112         // Create a sub message with only this empty parameter.
113         S7RequestMessage subMessage = new S7RequestMessage(
114             request.getMessageType(), (short) tpduRefGen.getAndIncrement(),
115             Collections.singletonList(subVarParameter), Collections.emptyList(), compositeRequestMessage);
116 
117         // Add this sub-message to the composite.
118         compositeRequestMessage.addRequestMessage(subMessage);
119 
120         // This calculates the size of the header for the request and response.
121         int initialRequestSize = S7RequestSizeCalculator.getRequestMessageSize(subMessage);
122         int curRequestSize = initialRequestSize;
123         int initialResponseSize = S7ResponseSizeEstimator.getEstimatedResponseMessageSize(subMessage);
124         int curResponseSize = initialResponseSize;
125 
126         VarParameter preProcessedVarParameter = new VarParameter(varParameter.getType(), new LinkedList<>());
127         for (VarParameterItem varParameterItem : varParameter.getItems()) {
128             // Use the S7RequestSizeCalculator to calculate the actual and estimated item sizes.
129             int itemRequestSize = S7RequestSizeCalculator.getRequestItemTotalSize(
130                 varParameterItem, null);
131             int itemResponseSize = S7ResponseSizeEstimator.getEstimatedResponseReadItemTotalSize(
132                 varParameterItem, null);
133 
134             // If the item would not fit into a separate message, we have to split it.
135             if((initialRequestSize + itemRequestSize > pduSize) || (initialResponseSize + itemResponseSize > pduSize)) {
136                 // The max response size is the size of the empty response, plus the type and num-items (each one byte) of one VarParameter, plus the size of the header one VarPayloadItem
137                 int maxResponseSize = pduSize - (initialResponseSize + 2 + 4);
138 
139                 S7AnyVarParameterItem s7AnyVarParameterItem = ((S7AnyVarParameterItem) varParameterItem);
140                 int maxNumElements = (int) Math.floor(
141                     (double) maxResponseSize / (double) s7AnyVarParameterItem.getDataType().getSizeInBytes());
142                 int sizeMaxNumElementInBytes = maxNumElements * s7AnyVarParameterItem.getDataType().getSizeInBytes();
143                 int remainingNumElements = s7AnyVarParameterItem.getNumElements();
144                 short curByteOffset = s7AnyVarParameterItem.getByteOffset();
145 
146                 while(remainingNumElements > 0) {
147                     int numCurElements = Math.min(remainingNumElements, maxNumElements);
148                     VarParameterItem subVarParameterItem = new S7AnyVarParameterItem(
149                         s7AnyVarParameterItem.getSpecificationType(), s7AnyVarParameterItem.getMemoryArea(),
150                         s7AnyVarParameterItem.getDataType(), numCurElements, s7AnyVarParameterItem.getDataBlockNumber(),
151                         curByteOffset, (byte) 0);
152                     preProcessedVarParameter.getItems().add(subVarParameterItem);
153 
154                     remainingNumElements -= maxNumElements;
155                     curByteOffset += sizeMaxNumElementInBytes;
156                 }
157             }
158             // In all other cases, just forward the item.
159             else {
160                 preProcessedVarParameter.getItems().add(varParameterItem);
161             }
162         }
163 
164         // For each var item of the original request, try adding them to the current sub-message
165         // as long as it or the resulting response does not exceed the max PDU size.
166         for (VarParameterItem varParameterItem : preProcessedVarParameter.getItems()) {
167             // Use the S7RequestSizeCalculator to calculate the actual and estimated item sizes.
168             int itemRequestSize = S7RequestSizeCalculator.getRequestItemTotalSize(
169                 varParameterItem, null);
170             int itemResponseSize = S7ResponseSizeEstimator.getEstimatedResponseReadItemTotalSize(
171                 varParameterItem, null);
172 
173             // If adding this item, would exceed either the request or response size,
174             // create a new sub-message and add this item to that.
175             if ((curRequestSize + itemRequestSize > pduSize) || (curResponseSize + itemResponseSize > pduSize)) {
176                 // Create a new var parameter without any items (yet).
177                 subVarParameter = new VarParameter(varParameter.getType(), new LinkedList<>());
178 
179                 // Create a new sub message with only this empty parameter.
180                 subMessage = new S7RequestMessage(
181                     request.getMessageType(), (short) tpduRefGen.getAndIncrement(),
182                     Collections.singletonList(subVarParameter),
183                     Collections.emptyList(), compositeRequestMessage);
184 
185                 // Reset the message size
186                 curRequestSize = S7RequestSizeCalculator.getRequestMessageSize(subMessage) + itemRequestSize;
187                 curResponseSize = S7ResponseSizeEstimator.getEstimatedResponseMessageSize(subMessage) + itemResponseSize;
188 
189                 // Add this new sub-message to the composite.
190                 compositeRequestMessage.addRequestMessage(subMessage);
191             } else {
192                 // Increment the current sizes.
193                 curRequestSize += itemRequestSize;
194                 curResponseSize += itemResponseSize;
195             }
196 
197             subVarParameter.getItems().add(varParameterItem);
198         }
199         return compositeRequestMessage;
200     }

202     private S7CompositeRequestMessage processWriteVarParameter(S7RequestMessage request, VarParameter varParameter)
203             throws PlcProtocolException {
204         // Create a new composite request message.
205         S7CompositeRequestMessage compositeRequestMessage = new S7CompositeRequestMessage(request);
206 
207         VarPayload varPayload = request.getPayload(VarPayload.class)
208             .orElseThrow(() -> new PlcProtocolException("Expecting payloads for a write request"));
209         if(varParameter.getItems().size() != varPayload.getItems().size()) {
210             throw new PlcProtocolException("Number of items in parameter and payload don't match");
211         }
212         List<VarParameterItem> parameterItems = varParameter.getItems();
213         List<VarPayloadItem> payloadItems = varPayload.getItems();
214 
215         for (int i = 0; i < parameterItems.size(); i++) {
216             VarParameterItem varParameterItem = parameterItems.get(i);
217             VarPayloadItem varPayloadItem = payloadItems.get(i);
218 
219             if (varParameterItem instanceof S7AnyVarParameterItem) {
220                 S7AnyVarParameterItem s7AnyVarParameterItem = (S7AnyVarParameterItem) varParameterItem;
221                 short byteOffset = s7AnyVarParameterItem.getByteOffset();
222                 if (s7AnyVarParameterItem.getDataType() == TransportSize.BOOL) {
223                     processBooleanWriteVarParameter(request, varParameter, varPayload, s7AnyVarParameterItem,
224                         varPayloadItem, byteOffset, compositeRequestMessage);
225                 } else {
226                     processNonBooleanWriteVarParameter(request, varParameter, varPayload, s7AnyVarParameterItem,
227                         varPayloadItem, byteOffset, compositeRequestMessage);
228                 }
229             } else {
230                 throw new NotImplementedException("Handling of other element types not implemented.");
231             }
232         }
233         return compositeRequestMessage;
234     }

236     private void processBooleanWriteVarParameter(S7RequestMessage request, VarParameter varParameter, VarPayload varPayload,
237                                          S7AnyVarParameterItem s7AnyVarParameterItem, VarPayloadItem varPayloadItem,
238                                          short byteOffset, S7CompositeRequestMessage compositeRequestMessage) {
239         short curParameterByteOffset = byteOffset;
240         byte curParameterBitOffset = s7AnyVarParameterItem.getBitOffset();
241         byte curPayloadBitOffset = 0;
242         for (int i = 0; i < s7AnyVarParameterItem.getNumElements(); i++) {
243             // Create a new message with only one single value item in the var parameter.
244             VarParameterItem item = new S7AnyVarParameterItem(
245                 s7AnyVarParameterItem.getSpecificationType(),
246                 s7AnyVarParameterItem.getMemoryArea(),
247                 s7AnyVarParameterItem.getDataType(), (short) 1,
248                 s7AnyVarParameterItem.getDataBlockNumber(),
249                 curParameterByteOffset, curParameterBitOffset);
250             S7Parameter subVarParameter = new VarParameter(varParameter.getType(),
251                 Collections.singletonList(item));
252 
253             // Create a one-byte byte array and set it to 0x01, if the corresponding bit
254             // was 1 else set it to 0x00.
255             byte[] elementData = new byte[1];
256             short curPayloadByteOffset = (short) ((short) i / 8);
257             elementData[0] = (byte) ((varPayloadItem.getData()[curPayloadByteOffset] >> curPayloadBitOffset) & 0x01);
258 
259             // Create the new payload item.
260             VarPayloadItem itemPayload = new VarPayloadItem(
261                 varPayloadItem.getReturnCode(),
262                 varPayloadItem.getDataTransportSize(), elementData);
263             S7Payload subVarPayload = new VarPayload(varPayload.getType(),
264                 Collections.singletonList(itemPayload));
265 
266             // Create a new sub message.
267             S7RequestMessage subMessage = new S7RequestMessage(
268                 request.getMessageType(), (short) tpduRefGen.getAndIncrement(),
269                 Collections.singletonList(subVarParameter),
270                 Collections.singletonList(subVarPayload),
271                 compositeRequestMessage);
272 
273             // Add the new message to the composite.
274             compositeRequestMessage.addRequestMessage(subMessage);
275 
276             // Calculate the new memory and bit offset.
277             curParameterBitOffset++;
278             if ((i > 0) && ((curParameterBitOffset % 8) == 0)) {
279                 curParameterByteOffset++;
280                 curParameterBitOffset = 0;
281             }
282             curPayloadBitOffset++;
283             if ((i > 0) && ((curPayloadBitOffset % 8) == 0)) {
284                 curPayloadBitOffset = 0;
285             }
286         }
287     }

289     private void processNonBooleanWriteVarParameter(S7RequestMessage request, VarParameter varParameter, VarPayload varPayload,
290                                             S7AnyVarParameterItem s7AnyVarParameterItem, VarPayloadItem varPayloadItem,
291                                             short byteOffset, S7CompositeRequestMessage compositeRequestMessage) {
292         short curByteOffset = byteOffset;
293         int payloadPosition = 0;
294         for (int i = 0; i < s7AnyVarParameterItem.getNumElements(); i++) {
295             int elementSize = s7AnyVarParameterItem.getDataType().getSizeInBytes();
296 
297             // Create a new message with only one single value item in the var parameter.
298             VarParameterItem itemParameter = new S7AnyVarParameterItem(
299                 s7AnyVarParameterItem.getSpecificationType(),
300                 s7AnyVarParameterItem.getMemoryArea(),
301                 s7AnyVarParameterItem.getDataType(), (short) 1,
302                 s7AnyVarParameterItem.getDataBlockNumber(),
303                 curByteOffset, (byte) 0);
304             S7Parameter subVarParameter = new VarParameter(varParameter.getType(),
305                 Collections.singletonList(itemParameter));
306 
307             // Split up the big array into a separate byte-array that contains a single element.
308             byte[] elementData = new byte[elementSize];
309             System.arraycopy(varPayloadItem.getData(), payloadPosition, elementData, 0, elementSize);
310             payloadPosition += elementSize;
311 
312             // Create the new payload item.
313             VarPayloadItem itemPayload = new VarPayloadItem(
314                 varPayloadItem.getReturnCode(),
315                 varPayloadItem.getDataTransportSize(), elementData);
316             S7Payload subVarPayload = new VarPayload(varPayload.getType(),
317                 Collections.singletonList(itemPayload));
318 
319             // Create a new sub message.
320             S7RequestMessage subMessage = new S7RequestMessage(
321                 request.getMessageType(), (short) tpduRefGen.getAndIncrement(),
322                 Collections.singletonList(subVarParameter),
323                 Collections.singletonList(subVarPayload),
324                 compositeRequestMessage);
325 
326             // Add the new message to the composite.
327             compositeRequestMessage.addRequestMessage(subMessage);
328 
329             // Calculate the new memory offset.
330             curByteOffset += elementSize;
331         }
332     }

334     @Override
335     public S7ResponseMessage processResponse(S7RequestMessage request, S7ResponseMessage response) {
336         // If it's a split-up message, check if all parts are now acknowledged.
337         if (request.getParent() instanceof S7CompositeRequestMessage) {
338             S7CompositeRequestMessage parent = (S7CompositeRequestMessage) request.getParent();
339 
340             // Add the response to the container so we can add it's information to the composite response.
341             parent.addResponseMessage(response);
342 
343             // If all parts of this split-up message are now acknowledged, create a unified
344             // response object and pass that up to the higher layers.
345             if (parent.isAcknowledged()) {
346                 return getMergedResponseMessage(parent.originalRequest, parent.getResponseMessages());
347             } else {
348                 return null;
349             }
350         }
351         // If it's an ordinary request, simply forward it  to the higher layers.
352         else {
353             return response;
354         }
355     }

357     private S7ResponseMessage getMergedResponseMessage(S7RequestMessage requestMessage,
358                                                        Collection<? extends S7ResponseMessage> responses) {
359         MessageType messageType = null;
360         short tpduReference = requestMessage.getTpduReference();
361         List<S7Parameter> s7Parameters = new LinkedList<>();
362         List<S7Payload> s7Payloads = new LinkedList<>();
363 
364         Optional<VarParameter> varParameterOptional = requestMessage.getParameter(VarParameter.class);
365 
366         // This is neither a read request nor a write request, just merge all parameters together.
367         if(!varParameterOptional.isPresent()) {
368             for (S7ResponseMessage response : responses) {
369                 messageType = response.getMessageType();
370                 s7Parameters.addAll(response.getParameters());
371                 s7Payloads.addAll(response.getPayloads());
372             }
373         }
374 
375         // This is a read or write request, we have to merge all the items in the var parameter.
376         else {
377             List<VarParameterItem> parameterItems = new LinkedList<>();
378             List<VarPayloadItem> payloadItems = new LinkedList<>();
379             for (S7ResponseMessage response : responses) {
380                 messageType = response.getMessageType();
381                 parameterItems.addAll(response.getParameter(VarParameter.class)
382                     .orElseThrow(() -> new PlcRuntimeException(
383                         "Every response of a Read message should have a VarParameter instance")).getItems());
384                 Optional<VarPayload> payload = response.getPayload(VarPayload.class);
385                 payload.ifPresent(varPayload -> payloadItems.addAll(varPayload.getItems()));
386             }
387 
388             List<VarParameterItem> mergedParameterItems = new LinkedList<>();
389             List<VarPayloadItem> mergedPayloadItems = new LinkedList<>();
390             VarParameter varParameter = varParameterOptional.get();
391 
392             int responseOffset = 0;
393             for(int i = 0; i < varParameter.getItems().size(); i++) {
394                 S7AnyVarParameterItem requestItem = (S7AnyVarParameterItem) varParameter.getItems().get(i);
395 
396                 // Get the pairs of corresponding parameter and payload items.
397                 S7AnyVarParameterItem responseParameterItem = (S7AnyVarParameterItem) parameterItems.get(0);
398                 VarPayloadItem responsePayloadItem = payloadItems.get(i + responseOffset);
399                 int dataOffset = (responsePayloadItem.getData() != null) ? responsePayloadItem.getData().length : 0;
400 
401                 // The resulting parameter items is identical to the request parameter item.
402                 mergedParameterItems.add(requestItem);
403 
404                 // The payload will have to be merged and the return codes will have to be examined.
405                 if(requestItem.getNumElements() != responseParameterItem.getNumElements()) {
406                     int itemSizeInBytes = requestItem.getDataType().getSizeInBytes();
407                     int totalSizeInBytes = requestItem.getNumElements() * itemSizeInBytes;
408                     byte[] data = null;
409                     if(varParameter.getType() == ParameterType.READ_VAR) {
410                         data = new byte[totalSizeInBytes];
411                         System.arraycopy(responsePayloadItem.getData(), 0, data, 0, responsePayloadItem.getData().length);
412                     }
413 
414                     // Initialize the current size, this will be lower than the original, as the only
415                     // way to have different count, is if the request was split up.
416                     int curSizeInBytes = responseParameterItem.getNumElements() * itemSizeInBytes;
417 
418                     // Now iterate over the succeeding pairs of parameters and payloads till we have
419                     // found the original number of elements.
420                     while(curSizeInBytes < totalSizeInBytes) {
421                         responseOffset++;
422                         // No need to process the parameters, we only need them to get the number of items.
423                         responseParameterItem = (S7AnyVarParameterItem) parameterItems.get(i + responseOffset);
424                         curSizeInBytes += responseParameterItem.getNumElements() * itemSizeInBytes;
425 
426                         // Get the next payload item in the list.
427                         responsePayloadItem = payloadItems.get(i + responseOffset);
428 
429                         // Copy the data of this item behind the previous content.
430                         if(varParameter.getType() == ParameterType.READ_VAR) {
431                             System.arraycopy(responsePayloadItem.getData(), 0, data, dataOffset, responsePayloadItem.getData().length);
432                             dataOffset += responsePayloadItem.getData().length;
433                         }
434                     }
435 
436                     mergedPayloadItems.add(new VarPayloadItem(DataTransportErrorCode.OK,
437                         responsePayloadItem.getDataTransportSize(), data));
438                 } else {
439                     mergedPayloadItems.add(responsePayloadItem);
440                 }
441             }
442 
443             s7Parameters.add(new VarParameter(varParameter.getType(), mergedParameterItems));
444             s7Payloads.add(new VarPayload(varParameter.getType(), mergedPayloadItems));
445         }
446         // TODO: The error codes are wrong
447         return new S7ResponseMessage(messageType, tpduReference, s7Parameters, s7Payloads, (byte) 0xFF, (byte) 0xFF);
448     }

450     static class S7CompositeRequestMessage implements PlcProtocolMessage {
451 
452         private S7RequestMessage originalRequest;
453         private List<S7RequestMessage> requestMessages;
454         private List<S7ResponseMessage> responseMessages;
455 
456         S7CompositeRequestMessage(S7RequestMessage originalRequest) {
457             this.originalRequest = originalRequest;
458             this.requestMessages = new LinkedList<>();
459             this.responseMessages = new LinkedList<>();
460         }
461 
462         @Override
463         public PlcProtocolMessage getParent() {
464             return originalRequest;
465         }
466 
467         /**
468          * A {@link S7CompositeRequestMessage} is only acknowledged, if all children are acknowledged.
469          *
470          * @return true if all children are acknowledged.
471          */
472         private boolean isAcknowledged() {
473             for (S7RequestMessage requestMessage : requestMessages) {
474                 if(!requestMessage.isAcknowledged()) {
475                     return false;
476                 }
477             }
478             return true;
479         }
480 
481         void addRequestMessage(S7RequestMessage requestMessage) {
482             requestMessages.add(requestMessage);
483         }
484 
485         public List<S7RequestMessage> getRequestMessages() {
486             return requestMessages;
487         }
488 
489         private void addResponseMessage(S7ResponseMessage responseMessage) {
490             responseMessages.add(responseMessage);
491         }
492 
493         public List<S7ResponseMessage> getResponseMessages() {
494             return responseMessages;
495         }
496     }
