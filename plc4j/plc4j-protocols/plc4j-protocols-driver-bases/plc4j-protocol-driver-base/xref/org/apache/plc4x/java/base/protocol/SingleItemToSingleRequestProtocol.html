108     public SingleItemToSingleRequestProtocol(PlcReader reader, PlcWriter writer, PlcSubscriber subscriber, Timer timer, long defaultReceiveTimeout, SplitConfig splitConfig, boolean betterImplementationPossible) {
109         this.reader = reader;
110         this.writer = writer;
111         this.subscriber = subscriber;
112         this.timer = timer;
113         this.defaultReceiveTimeout = defaultReceiveTimeout;
114         this.splitConfig = splitConfig;
115         if (this.splitConfig == null) {
116             this.splitConfig = new SplitConfig();
117         }
118         if (betterImplementationPossible) {
119             String callStack = Arrays.stream(Thread.currentThread().getStackTrace())
120                 .map(StackTraceElement::toString)
121                 .collect(Collectors.joining("\n"));
122             LOGGER.warn("Unoptimized Usage of {} detected at:\n{}", this.getClass(), callStack);
123             LOGGER.info("Consider implementing item splitting native to the protocol.");
124         }
125     }

181     protected void tryFinish(Integer currentTdpu, InternalPlcResponse msg, CompletableFuture<InternalPlcResponse> originalResponseFuture) {
182         deliveredItems.incrementAndGet();
183         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> subPlcRequestContainer = sentButUnacknowledgedSubContainer.remove(currentTdpu);
184         LOGGER.info("{} got acknowledged", subPlcRequestContainer);
185         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> originalPlcRequestContainer = correlationToParentContainer.remove(currentTdpu);
186         if (originalPlcRequestContainer == null) {
187             LOGGER.warn("Unrelated package received {}", msg);
188             return;
189         }
190         Queue<InternalPlcResponse> correlatedResponseItems = responsesToBeDelivered.computeIfAbsent(originalPlcRequestContainer, ignore -> new ConcurrentLinkedQueue<>());
191         correlatedResponseItems.add(msg);
192         Set<Integer> integers = containerCorrelationIdMap.get(originalPlcRequestContainer);
193         integers.remove(currentTdpu);
194         if (integers.isEmpty()) {
195             deliveredContainers.incrementAndGet();
196             Timeout timeout = scheduledTimeouts.remove(originalPlcRequestContainer);
197             if (timeout != null) {
198                 timeout.cancel();
199             }
200 
201             InternalPlcResponse plcResponse;
202             if (originalPlcRequestContainer.getRequest() instanceof InternalPlcReadRequest) {
203                 InternalPlcReadRequest internalPlcReadRequest = (InternalPlcReadRequest) originalPlcRequestContainer.getRequest();
204                 HashMap<String, Pair<PlcResponseCode, BaseDefaultFieldItem>> fields = new HashMap<>();
205 
206                 correlatedResponseItems.stream()
207                     .map(InternalPlcReadResponse.class::cast)
208                     .map(InternalPlcReadResponse::getValues)
209                     .forEach(stringPairMap -> stringPairMap.forEach(fields::put));
210 
211                 plcResponse = new DefaultPlcReadResponse(internalPlcReadRequest, fields);
212             } else if (originalPlcRequestContainer.getRequest() instanceof InternalPlcWriteRequest) {
213                 InternalPlcWriteRequest internalPlcWriteRequest = (InternalPlcWriteRequest) originalPlcRequestContainer.getRequest();
214                 HashMap<String, PlcResponseCode> values = new HashMap<>();
215 
216                 correlatedResponseItems.stream()
217                     .map(InternalPlcWriteResponse.class::cast)
218                     .map(InternalPlcWriteResponse::getValues)
219                     .forEach(stringPairMap -> stringPairMap.forEach(values::put));
220 
221                 plcResponse = new DefaultPlcWriteResponse(internalPlcWriteRequest, values);
222             } else if (originalPlcRequestContainer.getRequest() instanceof InternalPlcSubscriptionRequest) {
223                 InternalPlcSubscriptionRequest internalPlcSubscriptionRequest = (InternalPlcSubscriptionRequest) originalPlcRequestContainer.getRequest();
224                 HashMap<String, Pair<PlcResponseCode, PlcSubscriptionHandle>> fields = new HashMap<>();
225 
226                 correlatedResponseItems.stream()
227                     .map(InternalPlcSubscriptionResponse.class::cast)
228                     .map(InternalPlcSubscriptionResponse::getValues)
229                     .forEach(stringPairMap -> stringPairMap.forEach(fields::put));
230 
231                 plcResponse = new DefaultPlcSubscriptionResponse(internalPlcSubscriptionRequest, fields);
232             } else if (originalPlcRequestContainer.getRequest() instanceof InternalPlcUnsubscriptionRequest) {
233                 InternalPlcUnsubscriptionRequest internalPlcUnsubscriptionRequest = (InternalPlcUnsubscriptionRequest) originalPlcRequestContainer.getRequest();
234                 plcResponse = new DefaultPlcUnsubscriptionResponse(internalPlcUnsubscriptionRequest);
235             } else {
236                 errored(currentTdpu, new PlcProtocolException("Unknown type detected " + originalPlcRequestContainer.getRequest().getClass()), originalResponseFuture);
237                 return;
238             }
239             responsesToBeDelivered.remove(originalPlcRequestContainer);
240             containerCorrelationIdMap.remove(originalPlcRequestContainer);
241             originalResponseFuture.complete(plcResponse);
242         }
243     }
244 
245     protected void errored(Integer currentTdpu, Throwable throwable, CompletableFuture<InternalPlcResponse> originalResponseFuture) {
246         erroredItems.incrementAndGet();
247         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> subPlcRequestContainer = sentButUnacknowledgedSubContainer.remove(currentTdpu);
248         LOGGER.info("{} got errored", subPlcRequestContainer);
249 
250 
251         PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> originalPlcRequestContainer = correlationToParentContainer.remove(currentTdpu);
252         if (originalPlcRequestContainer == null) {
253             LOGGER.warn("Unrelated error received tdpu:{}", currentTdpu, throwable);
254         } else {
255             erroredContainers.incrementAndGet();
256             Timeout timeout = scheduledTimeouts.remove(originalPlcRequestContainer);
257             if (timeout != null) {
258                 timeout.cancel();
259             }
260             responsesToBeDelivered.remove(originalPlcRequestContainer);
261 
262             Set<Integer> tdpus = containerCorrelationIdMap.remove(originalPlcRequestContainer);
263             if (tdpus != null) {
264                 tdpus.forEach(tdpu -> {
265                     // TODO: technically the other items didn't error so do we increment?
266                     //erroredItems.incrementAndGet();
267                     sentButUnacknowledgedSubContainer.remove(tdpu);
268                     correlationToParentContainer.remove(tdpu);
269                 });
270             }
271 
272             LOGGER.warn("PlcRequestContainer {} and correlationId {} failed ", correlationToParentContainer, currentTdpu, throwable);
273             originalResponseFuture.completeExceptionally(throwable);
274         }
275     }

281     @Override
282     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) throws Exception {
283         if (msg instanceof PlcRequestContainer) {
284             @SuppressWarnings("unchecked")
285             PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> in = (PlcRequestContainer<InternalPlcRequest, InternalPlcResponse>) msg;
286             Set<Integer> tdpus = containerCorrelationIdMap.computeIfAbsent(in, plcRequestContainer -> ConcurrentHashMap.newKeySet());
287 
288             Timeout timeout = timer.newTimeout(timeout_ -> handleTimeout(timeout_, in, tdpus, System.nanoTime()), defaultReceiveTimeout, TimeUnit.MILLISECONDS);
289             scheduledTimeouts.put(in, timeout);
290 
291             // Create a promise that has to be called multiple times.
292             PromiseCombiner promiseCombiner = new PromiseCombiner();
293             InternalPlcRequest request = in.getRequest();
294             if (request instanceof InternalPlcFieldRequest && (splitConfig.splitRead || splitConfig.splitWrite || splitConfig.splitSubscription)) {
295                 InternalPlcFieldRequest internalPlcFieldRequest = (InternalPlcFieldRequest) request;
296 
297                 if (internalPlcFieldRequest instanceof InternalPlcReadRequest && splitConfig.splitRead) {
298                     InternalPlcReadRequest internalPlcReadRequest = (InternalPlcReadRequest) internalPlcFieldRequest;
299                     internalPlcReadRequest.getNamedFields().forEach(field -> {
300                         ChannelPromise subPromise = new DefaultChannelPromise(promise.channel());
301 
302                         Integer tdpu = correlationIdGenerator.getAndIncrement();
303                         CompletableFuture<InternalPlcResponse> correlatedCompletableFuture = new CompletableFuture<>();
304                         // Important: don't chain to above as we want the above to be completed not the result of when complete
305                         correlatedCompletableFuture
306                             .thenApply(InternalPlcResponse.class::cast)
307                             .whenComplete((internalPlcResponse, throwable) -> {
308                                 if (throwable != null) {
309                                     errored(tdpu, throwable, in.getResponseFuture());
310                                 } else {
311                                     tryFinish(tdpu, internalPlcResponse, in.getResponseFuture());
312                                 }
313                             });
314                         PlcRequestContainer<CorrelatedPlcReadRequest, InternalPlcResponse> correlatedPlcRequestContainer = new PlcRequestContainer<>(CorrelatedPlcReadRequest.of(reader, field, tdpu), correlatedCompletableFuture);
315                         correlationToParentContainer.put(tdpu, in);
316                         queue.add(correlatedPlcRequestContainer, subPromise);
317                         if (!tdpus.add(tdpu)) {
318                             throw new IllegalStateException("AtomicInteger should not create duplicated ids: " + tdpu);
319                         }
320                         promiseCombiner.add((Future) subPromise);
321                     });
322                 } else if (internalPlcFieldRequest instanceof InternalPlcWriteRequest && splitConfig.splitWrite) {
323                     InternalPlcWriteRequest internalPlcWriteRequest = (InternalPlcWriteRequest) internalPlcFieldRequest;
324                     internalPlcWriteRequest.getNamedFieldTriples().forEach(fieldItemTriple -> {
325                         ChannelPromise subPromise = new DefaultChannelPromise(promise.channel());
326 
327                         Integer tdpu = correlationIdGenerator.getAndIncrement();
328                         CompletableFuture<InternalPlcResponse> correlatedCompletableFuture = new CompletableFuture<>()
329                             .thenApply(InternalPlcResponse.class::cast)
330                             .whenComplete((internalPlcResponse, throwable) -> {
331                                 if (throwable != null) {
332                                     errored(tdpu, throwable, in.getResponseFuture());
333                                 } else {
334                                     tryFinish(tdpu, internalPlcResponse, in.getResponseFuture());
335                                 }
336                             });
337                         PlcRequestContainer<CorrelatedPlcWriteRequest, InternalPlcResponse> correlatedPlcRequestContainer = new PlcRequestContainer<>(CorrelatedPlcWriteRequest.of(writer, fieldItemTriple, tdpu), correlatedCompletableFuture);
338                         correlationToParentContainer.put(tdpu, in);
339                         queue.add(correlatedPlcRequestContainer, subPromise);
340                         if (!tdpus.add(tdpu)) {
341                             throw new IllegalStateException("AtomicInteger should not create duplicated ids: " + tdpu);
342                         }
343                         promiseCombiner.add((Future) subPromise);
344                     });
345                 } else if (internalPlcFieldRequest instanceof InternalPlcSubscriptionRequest && splitConfig.splitSubscription) {
346                     InternalPlcSubscriptionRequest internalPlcSubscriptionRequest = (InternalPlcSubscriptionRequest) internalPlcFieldRequest;
347                     internalPlcSubscriptionRequest.getNamedSubscriptionFields().forEach(field -> {
348                         ChannelPromise subPromise = new DefaultChannelPromise(promise.channel());
349 
350                         Integer tdpu = correlationIdGenerator.getAndIncrement();
351                         CompletableFuture<InternalPlcResponse> correlatedCompletableFuture = new CompletableFuture<>();
352                         // Important: don't chain to above as we want the above to be completed not the result of when complete
353                         correlatedCompletableFuture
354                             .thenApply(InternalPlcResponse.class::cast)
355                             .whenComplete((internalPlcResponse, throwable) -> {
356                                 if (throwable != null) {
357                                     errored(tdpu, throwable, in.getResponseFuture());
358                                 } else {
359                                     tryFinish(tdpu, internalPlcResponse, in.getResponseFuture());
360                                 }
361                             });
362                         PlcRequestContainer<CorrelatedPlcSubscriptionRequest, InternalPlcResponse> correlatedPlcRequestContainer = new PlcRequestContainer<>(CorrelatedPlcSubscriptionRequest.of(subscriber, field, tdpu), correlatedCompletableFuture);
363                         correlationToParentContainer.put(tdpu, in);
364                         queue.add(correlatedPlcRequestContainer, subPromise);
365                         if (!tdpus.add(tdpu)) {
366                             throw new IllegalStateException("AtomicInteger should not create duplicated ids: " + tdpu);
367                         }
368                         promiseCombiner.add((Future) subPromise);
369                     });
370                 } else {
371                     throw new PlcProtocolException("Unmapped request type " + request.getClass());
372                 }
373             } else if (request instanceof InternalPlcUnsubscriptionRequest && splitConfig.splitUnsubscription) {
374                 InternalPlcUnsubscriptionRequest internalPlcUnsubscriptionRequest = (InternalPlcUnsubscriptionRequest) request;
375                 internalPlcUnsubscriptionRequest.getInternalPlcSubscriptionHandles().forEach(handle -> {
376                     ChannelPromise subPromise = new DefaultChannelPromise(promise.channel());
377 
378                     Integer tdpu = correlationIdGenerator.getAndIncrement();
379                     CompletableFuture<InternalPlcResponse> correlatedCompletableFuture = new CompletableFuture<>();
380                     // Important: don't chain to above as we want the above to be completed not the result of when complete
381                     correlatedCompletableFuture
382                         .thenApply(InternalPlcResponse.class::cast)
383                         .whenComplete((internalPlcResponse, throwable) -> {
384                             if (throwable != null) {
385                                 errored(tdpu, throwable, in.getResponseFuture());
386                             } else {
387                                 tryFinish(tdpu, internalPlcResponse, in.getResponseFuture());
388                             }
389                         });
390                     PlcRequestContainer<CorrelatedPlcUnsubscriptionRequest, InternalPlcResponse> correlatedPlcRequestContainer = new PlcRequestContainer<>(CorrelatedPlcUnsubscriptionRequest.of(subscriber, handle, tdpu), correlatedCompletableFuture);
391                     correlationToParentContainer.put(tdpu, in);
392                     queue.add(correlatedPlcRequestContainer, subPromise);
393                     if (!tdpus.add(tdpu)) {
394                         throw new IllegalStateException("AtomicInteger should not create duplicated ids: " + tdpu);
395                     }
396                     promiseCombiner.add((Future) subPromise);
397                 });
398             } else {
399                 ChannelPromise subPromise = new DefaultChannelPromise(promise.channel());
400                 queue.add(msg, subPromise);
401                 promiseCombiner.add((Future) subPromise);
402             }
403 
404             promiseCombiner.finish(promise);
405 
406             // Start sending the queue content.
407             trySendingMessages(ctx);
408         } else {
409             super.write(ctx, msg, promise);
410         }
411     }

417     @SuppressWarnings("unchecked")
418     protected synchronized void trySendingMessages(ChannelHandlerContext ctx) {
419         while (queue.size() > 0) {
420             // Get the RequestItem that is up next in the queue.
421             PlcRequestContainer currentItem = (PlcRequestContainer) queue.current();
422             InternalPlcRequest request = currentItem.getRequest();
423 
424             // Send the TPDU.
425             try {
426                 ChannelFuture channelFuture = queue.removeAndWrite();
427                 ctx.flush();
428                 if (channelFuture == null) {
429                     break;
430                 }
431             } catch (Exception e) {
432                 LOGGER.error("Error sending more queues messages", e);
433                 ctx.fireExceptionCaught(e);
434             }
435 
436             if (request instanceof CorrelatedPlcRequest) {
437                 CorrelatedPlcRequest correlatedPlcRequest = (CorrelatedPlcRequest) request;
438 
439                 // Add it to the list of sentButUnacknowledgedSubContainer.
440                 sentButUnacknowledgedSubContainer.put(correlatedPlcRequest.getTdpu(), currentItem);
441 
442                 LOGGER.debug("container with id {}sent: ", correlatedPlcRequest.getTdpu(), currentItem);
443             }
444         }
445         ctx.flush();
446     }

448     private void handleTimeout(Timeout timeout, PlcRequestContainer<InternalPlcRequest, InternalPlcResponse> in, Set<Integer> tdpus, long scheduledAt) {
449         if (timeout.isCancelled()) {
450             LOGGER.debug("container {} with timeout {} got canceled", in, timeout);
451             return;
452         }
453         LOGGER.warn("container {} timed out:{}", in, timeout);
454         erroredContainers.incrementAndGet();
455         responsesToBeDelivered.remove(in);
456         containerCorrelationIdMap.remove(in);
457         tdpus.forEach(tdpu -> {
458             erroredItems.incrementAndGet();
459             sentButUnacknowledgedSubContainer.remove(tdpu);
460             correlationToParentContainer.remove(tdpu);
461         });
462         in.getResponseFuture().completeExceptionally(new PlcTimeoutException(System.nanoTime() - scheduledAt));
463     }

554     // TODO: maybe export to jmx
555     public Map<String, Number> getStatistics() {
556         HashMap<String, Number> statistics = new HashMap<>();
557         statistics.put("queue", queue.size());
558         statistics.put("sentButUnacknowledgedSubContainer", sentButUnacknowledgedSubContainer.size());
559         statistics.put("correlationToParentContainer", correlationToParentContainer.size());

565         statistics.put("deliveredContainers", deliveredContainers.get());
