19  package org.apache.plc4x.java.base.messages;
20  
21  import org.apache.plc4x.java.api.messages.*;
22  import org.apache.plc4x.java.api.model.PlcConsumerRegistration;
23  import org.apache.plc4x.java.api.model.PlcSubscriptionHandle;
24  
25  import java.util.Collection;
26  import java.util.concurrent.CompletableFuture;
27  import java.util.function.Consumer;
28  
29  /**
30   * Interface implemented by all PlcConnections that are able to receive notifications from remote resources.
31   */
32  public interface PlcSubscriber {
33  
34      /**
35       * Subscribes to fields on the PLC.
36       *
37       * @param subscriptionRequest subscription request containing at least one subscription request item.
38       * @return subscription response containing a subscription response item for each subscription request item.
39       */
40      CompletableFuture<PlcSubscriptionResponse> subscribe(PlcSubscriptionRequest subscriptionRequest);
41  
42      /**
43       * Unsubscribes from fields on the PLC. For unsubscribing the unsubscription request uses the subscription
44       * handle returned as part of the subscription response item.
45       *
46       * @param unsubscriptionRequest unsubscription request containing at least one unsubscription request item.
47       * @return unsubscription response containing a unsubscription response item for each unsubscription request item.
48       */
49      CompletableFuture<PlcUnsubscriptionResponse> unsubscribe(PlcUnsubscriptionRequest unsubscriptionRequest);
50  
51      /**
52       * @param consumer
53       * @param handles
54       * @return TODO: document me
55       */
56      PlcConsumerRegistration register(Consumer<PlcSubscriptionEvent> consumer, Collection<PlcSubscriptionHandle> handles);
57  
58      /**
59       * // TODO: document me.
60       */
61      void unregister(PlcConsumerRegistration registration);
62  
63  }
