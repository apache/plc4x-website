35  
36  import static org.hamcrest.Matchers.*;
37  import static org.junit.Assert.assertThat;
38  import static org.mockito.Mockito.RETURNS_DEEP_STUBS;
39  import static org.mockito.Mockito.mock;
40  
41  @RunWith(Parameterized.class)
42  public class CommandFactoryMethodTest {
43  
44      @Parameterized.Parameter
45      public Class<?> clazz;
46  
47      @Parameterized.Parameters(name = "{index} {0}")
48      public static Collection<Object[]> data() {
49          return Stream.of(
50              AdsAbstractRequest.class,
51              AdsAbstractResponse.class,
52              AdsAddDeviceNotificationRequest.class,
53              AdsAddDeviceNotificationResponse.class,
54              AdsDeleteDeviceNotificationRequest.class,
55              AdsDeleteDeviceNotificationResponse.class,
56              AdsDeviceNotificationRequest.class,
57              AdsDeviceNotificationResponse.class,
58              AdsReadDeviceInfoRequest.class,
59              AdsReadDeviceInfoResponse.class,
60              AdsReadRequest.class,
61              AdsReadResponse.class,
62              AdsReadStateRequest.class,
63              AdsReadStateResponse.class,
64              AdsReadWriteRequest.class,
65              AdsReadWriteResponse.class,
66              AdsWriteControlRequest.class,
67              AdsWriteControlResponse.class,
68              AdsWriteRequest.class,
69              AdsWriteResponse.class,
70              UnknownCommand.class
71          ).map(clazz -> new Object[]{clazz}).collect(Collectors.toList());
72      }
73  
74      @Test
75      public void testOf() throws Exception {
76          List<Method> getters = Arrays
77              .stream(clazz.getDeclaredMethods())
78              .filter(method -> (
79                  method.getName().startsWith("get") || method.getName().startsWith("is"))
80                  && Modifier.isPublic(method.getModifiers())
81                  && method.getParameterCount() == 0)
82              .collect(Collectors.toList());
83          for (Method method : clazz.getDeclaredMethods()) {
84              if (!method.getName().equals("of")) {
85                  continue;
86              }
87              Object invoke = method.invoke(null, Arrays.stream(method.getParameterTypes()).map(aClass -> {
88                      if (aClass == Command.class) {
89                          return Command.INVALID;
90                      } else if (aClass == LengthSupplier[].class) {
91                          return new LengthSupplier[]{() -> 0};
92                      } else if (aClass == long.class) {
93                          return 1L;
94                      } else if (aClass == List.class) {
95                          return Collections.emptyList();
96                      } else {
97                          return mock(aClass, RETURNS_DEEP_STUBS);
98                      }
99                  }
100             ).toArray());
101             assertThat(invoke, notNullValue());
102             assertThat(invoke, instanceOf(clazz));
103             assertThat(invoke.toString(), not(isEmptyString()));
104             // Testing getters for the coverage (sonar)
105             for (Method getter : getters) {
106                 getter.invoke(invoke);
107             }
108         }
109     }
