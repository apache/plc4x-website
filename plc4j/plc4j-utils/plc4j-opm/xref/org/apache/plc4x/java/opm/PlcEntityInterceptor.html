53  /**
54   * Interceptor for dynamic functionality of @{@link PlcEntity}.
55   * Basically, its {@link #interceptGetter(Object, Method, Callable, String, PlcDriverManager, AliasRegistry, Map, Map)} method is called for each
56   * invocation of a method on a connected @{@link PlcEntity} and does then the dynamic part.
57   * <p>
58   * For those not too familiar with the JVM's dispatch on can roughly imagine the intercept method being a "regular"
59   * method on the "proxied" entity and all parameters of the intercept method could then be access to local fields.
60   *
61   * @author julian
62   */
63  public class PlcEntityInterceptor {
64

74      /**
75       * Basic Intersector for all methods on the proxy object.
76       * It checks if the invoked method is a getter and if so, only retrieves the requested field, forwarding to
77       * the {@link #fetchAndSetValueForGetter(Object, Method, PlcDriverManager, String, AliasRegistry, Map)} method.
78       * <p>
79       * If the field is no getter, then all fields are refreshed by calling {@link #refetchAllFields(Object, PlcDriverManager, String, AliasRegistry, Map)}
80       * and then, the method is invoked.
81       *
82       * @param proxy         Object to intercept
83       * @param method        Method that was intercepted
84       * @param callable      Callable to call the method after fetching the values
85       * @param address       Address of the plc (injected from private field)
86       * @param driverManager DriverManager instance to use (injected from private field)
87       * @return possible result of the original methods invocation
88       * @throws OPMException Problems with plc / proxying
89       */
90      @SuppressWarnings("unused")
91      @RuntimeType
92      public static Object interceptGetter(@This Object proxy, @Origin Method method, @SuperCall Callable<?> callable,
93                                           @FieldValue(PlcEntityManager.PLC_ADDRESS_FIELD_NAME) String address,
94                                           @FieldValue(PlcEntityManager.DRIVER_MANAGER_FIELD_NAME) PlcDriverManager driverManager,
95                                           @FieldValue(PlcEntityManager.ALIAS_REGISTRY) AliasRegistry registry,
96                                           @FieldValue(PlcEntityManager.LAST_FETCHED) Map<String, Instant> lastFetched,
97                                           @FieldValue(PlcEntityManager.LAST_WRITTEN) Map<String, Instant> lastWritten) throws OPMException {
98          LOGGER.trace("Invoked method {} on connected PlcEntity {}", method.getName(), method.getDeclaringClass().getName());
99  
100         // If "detached" (i.e. _driverManager is null) simply forward the call
101         if (driverManager == null) {
102             LOGGER.trace("Entity not connected, simply fowarding call");
103             try {
104                 return callable.call();
105             } catch (Exception e) {
106                 throw new OPMException("Exception during forwarding call", e);
107             }
108         }
109 
110         if (method.getName().startsWith("get")) {
111             if (method.getParameterCount() > 0) {
112                 throw new OPMException("Only getter with no arguments are supported");
113             }
114             // Fetch single value
115             LOGGER.trace("Invoked method {} is getter, trying to find annotated field and return requested value",
116                 method.getName());
117 
118             fetchAndSetValueForGetter(proxy, method, driverManager, address, registry, lastFetched);
119             try {
120                 return callable.call();
121             } catch (Exception e) {
122                 throw new OPMException("Unable to forward invocation " + method.getName() + " on connected PlcEntity", e);
123             }
124         }
125 
126         if (method.getName().startsWith("is") && (method.getReturnType() == boolean.class || method.getReturnType() == Boolean.class)) {
127             if (method.getParameterCount() > 0) {
128                 throw new OPMException("Only getter with no arguments are supported");
129             }
130             // Fetch single value
131             LOGGER.trace("Invoked method {} is boolean flag method, trying to find annotated field and return requested value",
132                 method.getName());
133             fetchAndSetValueForIsGetter(proxy, method, driverManager, address, registry, lastFetched);
134             try {
135                 return callable.call();
136             } catch (Exception e) {
137                 throw new OPMException("Unable to forward invocation " + method.getName() + " on connected PlcEntity", e);
138             }
139         }
140 
141         // Fetch all values, than invoke method
142         try {
143             LOGGER.trace("Invoked method is no getter, refetch all fields and invoke method {} then", method.getName());
144             refetchAllFields(proxy, driverManager, address, registry, lastFetched);
145             Object call = callable.call();
146             // We write back
147             writeAllFields(proxy, driverManager, address, registry, lastWritten);
148             return call;
149         } catch (Exception e) {
150             throw new OPMException("Unable to forward invocation " + method.getName() + " on connected PlcEntity", e);
151         }
152     }

154     @RuntimeType
155     public static Object interceptSetter(@This Object proxy, @Origin Method method, @SuperCall Callable<?> callable,
156                                          @FieldValue(PlcEntityManager.PLC_ADDRESS_FIELD_NAME) String address,
157                                          @FieldValue(PlcEntityManager.DRIVER_MANAGER_FIELD_NAME) PlcDriverManager driverManager,
158                                          @FieldValue(PlcEntityManager.ALIAS_REGISTRY) AliasRegistry registry,
159                                          @FieldValue(PlcEntityManager.LAST_FETCHED) Map<String, Instant> lastFetched,
160                                          @Argument(0) Object argument) throws OPMException {
161         LOGGER.trace("Invoked method {} on connected PlcEntity {}", method.getName(), method.getDeclaringClass().getName());
162 
163         // If "detached" (i.e. _driverManager is null) simply forward the call
164         if (driverManager == null) {
165             LOGGER.trace("Entity not connected, simply fowarding call");
166             try {
167                 return callable.call();
168             } catch (Exception e) {
169                 throw new OPMException("Exception during forwarding call", e);
170             }
171         }
172 
173         if (method.getName().startsWith("set")) {
174             if (method.getParameterCount() != 1) {
175                 throw new OPMException("Only setter with one arguments are supported");
176             }
177             // Set single value
178             LOGGER.trace("Invoked method {} is setter, trying to find annotated field and return requested value",
179                 method.getName());
180 
181             return setValueForSetter(proxy, method, callable, driverManager, address, registry, lastFetched, argument);
182         }
183 
184         // Fetch all values, than invoke method
185         try {
186             LOGGER.trace("Invoked method is no getter, refetch all fields and invoke method {} then", method.getName());
187             refetchAllFields(proxy, driverManager, address, registry, lastFetched);
188             return callable.call();
189         } catch (Exception e) {
190             throw new OPMException("Unable to forward invocation " + method.getName() + " on connected PlcEntity", e);
191         }
192     }

194     /**
195      * Renews all values of all Fields that are annotated with {@link PlcEntity}.
196      *
197      * @param proxy         Object to refresh the fields on.
198      * @param driverManager Driver Manager to use
199      * @param registry      AliasRegistry to use
200      * @param lastFetched
201      * @throws OPMException on various errors.
202      */
203     @SuppressWarnings("squid:S1141") // Nested try blocks readability is okay, move to other method makes it imho worse
204     static void refetchAllFields(Object proxy, PlcDriverManager driverManager, String address, AliasRegistry registry, Map<String, Instant> lastFetched) throws OPMException {
205         // Don't log o here as this would cause a second request against a plc so don't touch it, or if you log be aware of that
206         Class<?> entityClass = proxy.getClass().getSuperclass();
207         LOGGER.trace("Refetching all fields on proxy object of class {}", entityClass);
208         PlcEntity plcEntity = entityClass.getAnnotation(PlcEntity.class);
209         if (plcEntity == null) {
210             throw new OPMException("Non PlcEntity supplied");
211         }
212 
213         // Check if all fields are valid
214         for (Field field : entityClass.getDeclaredFields()) {
215             if (field.isAnnotationPresent(PlcField.class)) {
216                 OpmUtils.getOrResolveAddress(registry, field.getAnnotation(PlcField.class).value());
217             }
218         }
219         try (PlcConnection connection = driverManager.getConnection(address)) {
220             // Catch the exception, if no reader present (see below)
221             // Build the query
222             PlcReadRequest.Builder requestBuilder = connection.readRequestBuilder();
223 
224             Arrays.stream(entityClass.getDeclaredFields())
225                 .filter(field -> field.isAnnotationPresent(PlcField.class))
226                 .filter(field -> needsToBeSynced(lastFetched, field))
227                 .forEach(field ->
228                     requestBuilder.addItem(
229                         getFqn(field),
230                         OpmUtils.getOrResolveAddress(registry, field.getAnnotation(PlcField.class).value())
231                     )
232                 );
233 
234             PlcReadRequest request = requestBuilder.build();
235 
236             LOGGER.trace("Request for refetch of {} was build and is {}", entityClass, request);
237 
238             PlcReadResponse response = getPlcReadResponse(request);
239 
240             // Fill all requested fields
241             for (String fieldName : response.getFieldNames()) {
242                 // Fill into Cache
243                 lastFetched.put(fieldName, Instant.now());
244 
245                 LOGGER.trace("Value for field {}  is {}", fieldName, response.getObject(fieldName));
246                 String clazzFieldName = StringUtils.substringAfterLast(fieldName, ".");
247                 try {
248                     setField(entityClass, proxy, response, clazzFieldName, fieldName);
249                 } catch (NoSuchFieldException | IllegalAccessException e) {
250                     throw new PlcRuntimeException(e);
251                 }
252             }
253         } catch (PlcConnectionException e) {
254             throw new OPMException("Problem during processing", e);
255         } catch (Exception e) {
256             throw new OPMException("Unexpected error during processing", e);
257         }
258     }

260     static void writeAllFields(Object proxy, PlcDriverManager driverManager, String address, AliasRegistry registry, Map<String, Instant> lastWritten) throws OPMException {
261         // Don't log o here as this would cause a second request against a plc so don't touch it, or if you log be aware of that
262         Class<?> entityClass = proxy.getClass().getSuperclass();
263         LOGGER.trace("Writing all fields on proxy object of class {}", entityClass);
264         PlcEntity plcEntity = entityClass.getAnnotation(PlcEntity.class);
265         if (plcEntity == null) {
266             throw new OPMException("Non PlcEntity supplied");
267         }
268 
269         // Check if all fields are valid
270         for (Field field : entityClass.getDeclaredFields()) {
271             if (field.isAnnotationPresent(PlcField.class)) {
272                 OpmUtils.getOrResolveAddress(registry, field.getAnnotation(PlcField.class).value());
273             }
274         }
275         try (PlcConnection connection = driverManager.getConnection(address)) {
276             // Catch the exception, if no reader present (see below)
277             // Build the query
278             PlcWriteRequest.Builder requestBuilder = connection.writeRequestBuilder();
279 
280             Arrays.stream(entityClass.getDeclaredFields())
281                 .filter(field -> field.isAnnotationPresent(PlcField.class))
282                 .filter(field -> needsToBeSynced(lastWritten, field))
283                 .forEach(field ->
284                     requestBuilder.addItem(
285                         getFqn(field),
286                         OpmUtils.getOrResolveAddress(registry, field.getAnnotation(PlcField.class).value()),
287                         getFromField(field, proxy)
288                     )
289                 );
290 
291             PlcWriteRequest request = requestBuilder.build();
292 
293             LOGGER.trace("Request for write of {} was build and is {}", entityClass, request);
294 
295             PlcWriteResponse response = getPlcWriteResponse(request);
296 
297             // Fill all requested fields
298             for (String fieldName : response.getFieldNames()) {
299                 // Fill into Cache
300                 lastWritten.put(fieldName, Instant.now());
301             }
302         } catch (PlcConnectionException e) {
303             throw new OPMException("Problem during processing", e);
304         } catch (Exception e) {
305             throw new OPMException("Unexpected error during processing", e);
306         }
307     }

309     private static Object getFromField(Field field, Object object) {
310         try {
311             field.setAccessible(true);
312             return field.get(object);
313         } catch (IllegalAccessException e) {
314             throw new PlcRuntimeException(e);
315         }
316     }

318     private static String getFqn(Field field) {
319         return field.getDeclaringClass().getName() + "." + field.getName();
320     }

322     /**
323      * Checks if a field needs to be refetched/rewritten, i.e., the cached values are too old.
324      */
325     private static boolean needsToBeSynced(Map<String, Instant> lastSynced, Field field) {
326         Validate.notNull(field);
327         long cacheDurationMillis = field.getAnnotation(PlcField.class).cacheDurationMillis();
328         if (cacheDurationMillis < 0) {
329             return true;
330         }
331         String fqn = getFqn(field);
332         if (lastSynced.containsKey(fqn)) {
333             Instant last = lastSynced.get(fqn);
334             return Instant.now().minus(cacheDurationMillis, ChronoUnit.MILLIS).isAfter(last);
335         }
336         return true;
337     }

347     private static void fetchAndSetValueForGetter(Object proxy, Method m, int prefixLength, PlcDriverManager driverManager,
348                                                   String address, AliasRegistry registry, Map<String, Instant> lastFetched) throws OPMException {
349         String s = m.getName().substring(prefixLength);
350         // First char to lower
351         String variable = s.substring(0, 1).toLowerCase().concat(s.substring(1));
352         LOGGER.trace("Looking for field with name {} after invokation of getter {}", variable, m.getName());
353         PlcField annotation;
354         Field field;
355         try {
356             field = m.getDeclaringClass().getDeclaredField(variable);
357             annotation = field.getDeclaredAnnotation(PlcField.class);
358         } catch (NoSuchFieldException e) {
359             throw new OPMException("Unable to identify field with name '" + variable + "' for call to '" + m.getName() + "'", e);
360         }
361 
362         // Use Fully qualified Name as field index
363         String fqn = getFqn(field);
364 
365         // Check if cache is still active
366         if (!needsToBeSynced(lastFetched, field)) {
367             return;
368         }
369         try (PlcConnection connection = driverManager.getConnection(address)) {
370             // Catch the exception, if no reader present (see below)
371 
372             PlcReadRequest request = connection.readRequestBuilder()
373                 .addItem(fqn, OpmUtils.getOrResolveAddress(registry, annotation.value()))
374                 .build();
375 
376             PlcReadResponse response = getPlcReadResponse(request);
377 
378             // Fill into Cache
379             lastFetched.put(field.getName(), Instant.now());
380 
381             Object value = getTyped(m.getReturnType(), response, fqn);
382             setForField(field, proxy, value);
383         } catch (ClassCastException e) {
384             throw new OPMException("Unable to return response as suitable type", e);
385         } catch (Exception e) {
386             throw new OPMException("Problem during processing", e);
387         }
388     }

394         } catch (IllegalAccessException e) {
395             throw new PlcRuntimeException(e);

399     private static Object setValueForSetter(Object proxy, Method m, Callable<?> callable, PlcDriverManager driverManager,
400                                             String address, AliasRegistry registry, Map<String, Instant> lastFetched, Object object) throws OPMException {
401         String s = m.getName().substring(3);
402         // First char to lower
403         String variable = s.substring(0, 1).toLowerCase().concat(s.substring(1));
404         LOGGER.trace("Looking for field with name {} after invokation of getter {}", variable, m.getName());
405         PlcField annotation;
406         Field field;
407         try {
408             field = m.getDeclaringClass().getDeclaredField(variable);
409             annotation = field.getDeclaredAnnotation(PlcField.class);
410         } catch (NoSuchFieldException e) {
411             throw new OPMException("Unable to identify field with name '" + variable + "' for call to '" + m.getName() + "'", e);
412         }
413 
414         // Use Fully qualified Name as field index
415         String fqn = getFqn(field);
416 
417         try (PlcConnection connection = driverManager.getConnection(address)) {
418             // Catch the exception, if no reader present (see below)
419 
420             PlcWriteRequest request = connection.writeRequestBuilder()
421                 .addItem(fqn, OpmUtils.getOrResolveAddress(registry, annotation.value()), object)
422                 .build();
423 
424             PlcWriteResponse response = getPlcWriteResponse(request);
425 
426             // Fill into Cache
427             lastFetched.put(field.getName(), Instant.now());
428 
429             LOGGER.debug("getTyped clazz: {}, response: {}, fieldName: {}", m.getParameters()[0].getType(), response, fqn);
430             if (response.getResponseCode(fqn) != PlcResponseCode.OK) {
431                 throw new PlcRuntimeException(String.format("Unable to read specified field '%s', response code was '%s'",
432                     fqn, response.getResponseCode(fqn)));
433             }
434             callable.call();
435             return null;
436         } catch (ClassCastException e) {
437             throw new OPMException("Unable to return response as suitable type", e);
438         } catch (Exception e) {
439             throw new OPMException("Problem during processing", e);
440         }
441     }

444     /**
445      * Tries to set a response Item to a field in the given object.
446      * This is one by looking for a field in the class and a response item
447      * which is equal to the given fieldName parameter.
448      *
449      * @param o               Object to set the value on
450      * @param response        Response to fetch the response from
451      * @param targetFieldName Name of the field in the object
452      * @param sourceFieldName Name of the field in the response
453      * @throws NoSuchFieldException   If a field is not present in entity
454      * @throws IllegalAccessException If a field in the entity cannot be accessed
455      */
456     static void setField(Class<?> clazz, Object o, PlcReadResponse response, String targetFieldName, String sourceFieldName) throws NoSuchFieldException, IllegalAccessException {
457         LOGGER.debug("setField on clazz: {}, Object: {}, response: {}, targetFieldName: {}, sourceFieldName:{} ", clazz, o, response, targetFieldName, sourceFieldName);
458         Field field = clazz.getDeclaredField(targetFieldName);
459         field.setAccessible(true);
460         try {
461             field.set(o, getTyped(field.getType(), response, sourceFieldName));
462         } catch (ClassCastException e) {
463             throw new PlcRuntimeException(String.format("Unable to assign return value %s to field %s with type %s",
464                 response.getObject(sourceFieldName), targetFieldName, field.getType()), e);
465         }
466     }

468     @SuppressWarnings({"squid:S3776", "squid:MethodCyclomaticComplexity"})
469     // Cognitive Complexity not too high, as highly structured
470     static Object getTyped(Class<?> clazz, PlcReadResponse response, String sourceFieldName) {
471         LOGGER.debug("getTyped clazz: {}, response: {}, fieldName: {}", clazz, response, sourceFieldName);
472         if (response.getResponseCode(sourceFieldName) != PlcResponseCode.OK) {
473             throw new PlcRuntimeException(String.format("Unable to read specified field '%s', response code was '%s'",
474                 sourceFieldName, response.getResponseCode(sourceFieldName)));
475         }
476         if (clazz.isPrimitive()) {
477             if (clazz == boolean.class) {
478                 return response.getBoolean(sourceFieldName);
479             } else if (clazz == byte.class) {
480                 return response.getByte(sourceFieldName);
481             } else if (clazz == short.class) {
482                 return response.getShort(sourceFieldName);
483             } else if (clazz == int.class) {
484                 return response.getInteger(sourceFieldName);
485             } else if (clazz == long.class) {
486                 return response.getLong(sourceFieldName);
487             }
488         }
489 
490         if (clazz == Boolean.class) {
491             return response.getBoolean(sourceFieldName);
492         } else if (clazz == Byte.class) {
493             return response.getByte(sourceFieldName);
494         } else if (clazz == Short.class) {
495             return response.getShort(sourceFieldName);
496         } else if (clazz == Integer.class) {
497             return response.getInteger(sourceFieldName);
498         } else if (clazz == Long.class) {
499             return response.getLong(sourceFieldName);
500         } else if (clazz == BigInteger.class) {
501             return response.getBigInteger(sourceFieldName);
502         } else if (clazz == Float.class) {
503             return response.getFloat(sourceFieldName);
504         } else if (clazz == Double.class) {
505             return response.getDouble(sourceFieldName);
506         } else if (clazz == BigDecimal.class) {
507             return response.getBigDecimal(sourceFieldName);
508         } else if (clazz == String.class) {
509             return response.getString(sourceFieldName);
510         } else if (clazz == LocalTime.class) {
511             return response.getTime(sourceFieldName);
512         } else if (clazz == LocalDate.class) {
513             return response.getDate(sourceFieldName);
514         } else if (clazz == LocalDateTime.class) {
515             return response.getDateTime(sourceFieldName);
516         } else if (clazz == byte[].class) {
517             return ArrayUtils.toPrimitive(response.getByteArray(sourceFieldName));
518         } else if (clazz == Byte[].class) {
519             return response.getByteArray(sourceFieldName);
520         }
521 
522         // Fallback
523         Object responseObject = response.getObject(sourceFieldName);
524         if (clazz.isAssignableFrom(responseObject.getClass())) {
525             return responseObject;
526         }
527 
528         // If nothing matched, throw
529         throw new ClassCastException("Unable to return response item " + responseObject + "(" + responseObject.getClass() + ") as instance of " + clazz);
530     }

554     @SuppressWarnings("unchecked")
555     private static <REQ extends PlcRequest, RES extends PlcResponse> RES getFromFuture(REQ request) throws OPMException {
556         try {
557             return (RES) request.execute().get(READ_TIMEOUT, TimeUnit.MILLISECONDS);
558         } catch (InterruptedException e) {
559             Thread.currentThread().interrupt();
560             throw new OPMException("Exception during execution", e);
561         } catch (ExecutionException e) {
562             throw new OPMException("Exception during execution", e);
563         } catch (TimeoutException e) {
564             throw new OPMException("Timeout during fetching values", e);
