50      @Test
51      public void read_throwsInvalidFieldException_rethrows() throws PlcConnectionException {
52          // Prepare the Mock
53          PlcDriverManager driverManager = Mockito.mock(PlcDriverManager.class);
54          PlcConnection connection = Mockito.mock(PlcConnection.class);
55          PlcConnectionMetadata metadata = Mockito.mock(PlcConnectionMetadata.class);
56          PlcReadRequest.Builder builder = Mockito.mock(PlcReadRequest.Builder.class);
57          when(metadata.canRead()).thenReturn(true);
58          when(connection.readRequestBuilder()).thenReturn(builder);
59          when(connection.getMetadata()).thenReturn(metadata);
60          when(builder.build()).thenThrow(new PlcInvalidFieldException("field1"));
61          when(driverManager.getConnection(any())).thenReturn(connection);
62  
63          // Create Entity Manager
64          PlcEntityManager entityManager = new PlcEntityManager(driverManager);
65  
66          // Issue Call to trigger interception
67          String message = null;
68          try {
69              BadEntity entity = entityManager.read(BadEntity.class, "mock:test");
70          } catch (Exception e) {
71              message = e.getMessage();
72          }
73  
74          assertEquals("Unable to parse field 'field1'", message);
75      }

101         PlcDriverManager driverManager = new PlcDriverManager();
102         PlcMockConnection connection = (PlcMockConnection) driverManager.getConnection("mock:test");

106                 // Sleep for 3s

109             }
110         });
111         connection.setDevice(mockDevice);
112 
113         // Create Entity Manager
114         PlcEntityManager entityManager = new PlcEntityManager(driverManager);
115 
116         // Issue Call which SHOULD timeout
117         BadEntity entity = entityManager.read(BadEntity.class, "mock:test");

138     @Test
139     public void read_resolveAlias_works() throws OPMException, PlcConnectionException {
140         SimpleAliasRegistry registry = new SimpleAliasRegistry();
141         registry.register("alias", "real_field");
142 
143         // Mock
144         PlcDriverManager driverManager = new PlcDriverManager();
145         PlcMockConnection connection = (PlcMockConnection) driverManager.getConnection("mock:test");
146         MockDevice mockDevice = Mockito.mock(MockDevice.class);
147         when(mockDevice.read(any())).thenReturn(Pair.of(PlcResponseCode.OK, new DefaultStringFieldItem("value")));
148         connection.setDevice(mockDevice);
149 
150         PlcEntityManager entityManager = new PlcEntityManager(driverManager, registry);
151         entityManager.read(AliasEntity.class, "mock:test");
152 
153         // Assert that "field" was queried
154         verify(mockDevice).read(eq("real_field"));
155     }
