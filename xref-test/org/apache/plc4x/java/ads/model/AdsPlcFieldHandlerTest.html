490     private static Stream<Arguments> createInputArrays() {
491         // Generate valid fields for each ads type.
492         Map<AdsDataType, PlcField> fields = new HashMap<>();
493         Arrays.stream(AdsDataType.values()).forEach(adsDataType -> {
494             String fieldQuery = "1/2:" + adsDataType.name();
495             fields.put(adsDataType, SUT.createField(fieldQuery));
496         });
497         // Generate output for each combination of ads and Java type.
498         Stream<Arguments> values = null;
499         for (AdsDataType adsDataType : AdsDataType.values()) {
500             PlcField field = fields.get(adsDataType);
501             for (InputTypes javaType : InputTypes.values()) {
502                 Object[] testValues = javaType.values;
503 
504                 BaseDefaultFieldItem fieldItem;
505                 try {
506                     fieldItem = javaType.fieldItemType.getDeclaredConstructor(Array.newInstance(testValues[0].getClass(), 0).getClass()).newInstance(new Object[]{testValues});
507                 } catch (InstantiationException | IllegalAccessException | InvocationTargetException | NoSuchMethodException e) {
508                     throw new PlcRuntimeException("Error initializing field class " + javaType.fieldItemType.getSimpleName(), e);
509                 }
510 
511                 Stream<Arguments> curValues;
512                 // Min, Max
513                 if (testValues.length == 2) {
514                     curValues = Stream.of(
515                         Arguments.of(adsDataType.name() + "-" + javaType.name() + "-MIN", field, createOneElementArray(testValues[0])),
516                         Arguments.of(adsDataType.name() + "-" + javaType.name() + "-MAX", field, createOneElementArray(testValues[1])));
517                 }
518                 // Value, Min, Max
519                 else if (testValues.length == 3) {
520                     curValues = Stream.of(
521                         Arguments.of(adsDataType.name() + "-" + javaType.name() + "-MIN", field, createOneElementArray(testValues[1])),
522                         Arguments.of(adsDataType.name() + "-" + javaType.name() + "-VAL", field, createOneElementArray(testValues[0])),
523                         Arguments.of(adsDataType.name() + "-" + javaType.name() + "-MAX", field, createOneElementArray(testValues[2])));
524                 }
525                 // Zero, Value, Min, Max
526                 else if (testValues.length == 4) {
527                     curValues = Stream.of(
528                         Arguments.of(adsDataType.name() + "-" + javaType.name() + "-MIN", field, createOneElementArray(testValues[2])),
529                         Arguments.of(adsDataType.name() + "-" + javaType.name() + "-NIL", field, createOneElementArray(testValues[0])),
530                         Arguments.of(adsDataType.name() + "-" + javaType.name() + "-VAL", field, createOneElementArray(testValues[1])),
531                         Arguments.of(adsDataType.name() + "-" + javaType.name() + "-MAX", field, createOneElementArray(testValues[3])));
532                 } else {
533                     throw new RuntimeException("Expecting 2, 3 or 4 valued test-input");
534                 }
535                 if (values == null) {
536                     values = curValues;
537                 } else {
538                     values = Stream.concat(values, curValues);
539                 }
540             }
541         }
542         return values;
543     }

545     private static Object[] createOneElementArray(Object value) {
546         Class<?> type = value.getClass();
547         Object[] array = (Object[]) Array.newInstance(type, 1);
548         array[0] = value;
549         return array;
550     }

552     enum InputTypes {
553         BOOLEAN(DefaultBooleanFieldItem.class, new Boolean[]{false, true}),
554         BYTE(DefaultByteFieldItem.class, new Byte[]{(byte) 0, (byte) 42, Byte.MIN_VALUE, Byte.MAX_VALUE}),
555         SHORT(DefaultShortFieldItem.class, new Short[]{(short) 0, (short) 42, Short.MIN_VALUE, Short.MAX_VALUE}),
556         INTEGER(DefaultIntegerFieldItem.class, new Integer[]{0, 42, Integer.MIN_VALUE, Integer.MAX_VALUE}),
557         LONG(DefaultLongFieldItem.class, new Long[]{(long) 0, (long) 42, Long.MIN_VALUE, Long.MAX_VALUE}),
558         FLOAT(DefaultFloatFieldItem.class, new Float[]{(float) 0.0, (float) 42.23, -Float.MAX_VALUE, Float.MAX_VALUE}),
559         DOUBLE(DefaultDoubleFieldItem.class, new Double[]{0.0, 42.23, -Double.MAX_VALUE, Double.MAX_VALUE}),
560         // Creates an empty sting as min and a 254 char long string as max.
561         STRING(DefaultStringFieldItem.class, new String[]{"Hurz", "", IntStream.range(0, 127).mapToObj(i -> "a").collect(Collectors.joining(""))}),
562         TIME(DefaultLocalTimeFieldItem.class, new LocalTime[]{LocalTime.now(), LocalTime.MIDNIGHT, LocalTime.MIN, LocalTime.MAX}),
563         DATE(DefaultLocalDateFieldItem.class, new LocalDate[]{LocalDate.now(), LocalDate.MIN, LocalDate.MAX}),
564         DATETIME(DefaultLocalDateTimeFieldItem.class, new LocalDateTime[]{LocalDateTime.now(), LocalDateTime.MIN, LocalDateTime.MAX});

566         private final Class<? extends BaseDefaultFieldItem> fieldItemType;
567         private final Object[] values;

569         InputTypes(Class<? extends BaseDefaultFieldItem> fieldItemType, Object[] values) {
570             this.fieldItemType = fieldItemType;
571             this.values = values;
572         }

575     private void encode(String name, PlcField field, Object[] values, Set<String> expectedSuccess,
576                         BiFunction<PlcField, Object[], BaseDefaultFieldItem> encoder) {
577         boolean success = expectedSuccess.contains(name);
578         try {
579             BaseDefaultFieldItem fieldItem = encoder.apply(field, values);
580             assertNotNull(fieldItem, "A FieldItem instance should have been returned for testcase " + name);
581             if (!success) {
582                 fail(testInfo.getDisplayName() + ".\nExpected to fail for testcase " + name);
583             }
584         } catch (Exception e) {
585             if (success) {
586                 fail(testInfo.getDisplayName() + ".\nExpected to succeed for testcase " + name + " got exception " + e.getMessage(), e);
587             }
588         }
589     }
