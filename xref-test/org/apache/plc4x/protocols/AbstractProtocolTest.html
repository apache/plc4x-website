20  package org.apache.plc4x.protocols;
21  
22  import org.apache.daffodil.tdml.DFDLTestSuite;
23  import org.apache.daffodil.tdml.Runner;
24  import org.apache.daffodil.util.Misc;
25  import org.junit.jupiter.api.DynamicTest;
26  import org.junit.jupiter.api.TestFactory;
27  import scala.collection.Iterator;
28  
29  import java.util.LinkedList;
30  import java.util.List;
31  
32  public abstract class AbstractProtocolTest {
33  
34      private final String testsuiteName;
35  
36      public AbstractProtocolTest(String testsuiteName) {
37          this.testsuiteName = testsuiteName;
38      }
39  
40      @TestFactory
41      public List<DynamicTest> getTestsuiteTests() {
42          DFDLTestSuite testSuite = new DFDLTestSuite(Misc.getRequiredResource(testsuiteName), true, true, false,
43              Runner.defaultRoundTripDefaultDefault(), Runner.defaultValidationDefaultDefault());
44          List<DynamicTest> dynamicTests = new LinkedList<>();
45          Iterator<String> iterator = testSuite.testCaseMap().keySet().iterator();
46          while(iterator.hasNext()) {
47              String testcaseName = iterator.next();
48              String testcaseLabel = testSuite.suiteName() + ": " + testcaseName;
49              DynamicTest test = DynamicTest.dynamicTest(testcaseLabel, () ->
50                  testSuite.runOneTest(testcaseName, scala.Option.apply(null), false)
51              );
52              dynamicTests.add(test);
53          }
54          return dynamicTests;
55      }
56  
57  }
