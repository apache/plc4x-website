20  package org.apache.plc4x.java.examples.cloud.google;
21  
22  import org.apache.commons.cli.*;
23  
24  /** Command line options for the MQTT example. */
25  public class MqttExampleOptions {
26      String projectId;
27      String registryId;
28      String deviceId;
29      String privateKeyFile;
30      String algorithm;
31      String cloudRegion = "europe-west1";
32      int numMessages = 100;
33      int tokenExpMins = 20;
34      String mqttBridgeHostname = "mqtt.googleapis.com";
35      short mqttBridgePort = 8883;
36      String messageType = "event";
37  
38      /** Construct an MqttExampleOptions class from command line flags. */
39      public static MqttExampleOptions fromFlags(String[] args) {
40          Options options = new Options();
41          // Required arguments
42          options.addOption(
43              Option.builder()
44                  .type(String.class)
45                  .longOpt("project_id")
46                  .hasArg()
47                  .desc("GCP cloud project name.")
48                  .required()
49                  .build());
50          options.addOption(
51              Option.builder()
52                  .type(String.class)
53                  .longOpt("registry_id")
54                  .hasArg()
55                  .desc("Cloud IoT Core registry id.")
56                  .required()
57                  .build());
58          options.addOption(
59              Option.builder()
60                  .type(String.class)
61                  .longOpt("device_id")
62                  .hasArg()
63                  .desc("Cloud IoT Core device id.")
64                  .required()
65                  .build());
66          options.addOption(
67              Option.builder()
68                  .type(String.class)
69                  .longOpt("private_key_file")
70                  .hasArg()
71                  .desc("Path to private key file.")
72                  .required()
73                  .build());
74          options.addOption(
75              Option.builder()
76                  .type(String.class)
77                  .longOpt("algorithm")
78                  .hasArg()
79                  .desc("Encryption algorithm to use to generate the JWT. Either 'RS256' or 'ES256'.")
80                  .required()
81                  .build());
82  
83          // Optional arguments.
84          options.addOption(
85              Option.builder()
86                  .type(String.class)
87                  .longOpt("cloud_region")
88                  .hasArg()
89                  .desc("GCP cloud region.")
90                  .build());
91          options.addOption(
92              Option.builder()
93                  .type(Number.class)
94                  .longOpt("num_messages")
95                  .hasArg()
96                  .desc("Number of messages to publish.")
97                  .build());
98          options.addOption(
99              Option.builder()
100                 .type(String.class)
101                 .longOpt("mqtt_bridge_hostname")
102                 .hasArg()
103                 .desc("MQTT bridge hostname.")
104                 .build());
105         options.addOption(
106             Option.builder()
107                 .type(Number.class)
108                 .longOpt("token_exp_minutes")
109                 .hasArg()
110                 .desc("Minutes to JWT token refresh (token expiration time).")
111                 .build());
112         options.addOption(
113             Option.builder()
114                 .type(Number.class)
115                 .longOpt("mqtt_bridge_port")
116                 .hasArg()
117                 .desc("MQTT bridge port.")
118                 .build());
119         options.addOption(
120             Option.builder()
121                 .type(String.class)
122                 .longOpt("message_type")
123                 .hasArg()
124                 .desc("Indicates whether the message is a telemetry event or a device state message")
125                 .build());
126 
127         CommandLineParser parser = new DefaultParser();
128         CommandLine commandLine;
129         try {
130             commandLine = parser.parse(options, args);
131             MqttExampleOptions res = new MqttExampleOptions();
132 
133             res.projectId = commandLine.getOptionValue("project_id");
134             res.registryId = commandLine.getOptionValue("registry_id");
135             res.deviceId = commandLine.getOptionValue("device_id");
136             res.privateKeyFile = commandLine.getOptionValue("private_key_file");
137             res.algorithm = commandLine.getOptionValue("algorithm");
138             if (commandLine.hasOption("cloud_region")) {
139                 res.cloudRegion = commandLine.getOptionValue("cloud_region");
140             }
141             if (commandLine.hasOption("num_messages")) {
142                 res.numMessages = ((Number) commandLine.getParsedOptionValue("num_messages")).intValue();
143             }
144             if (commandLine.hasOption("token_exp_minutes")) {
145                 res.tokenExpMins =
146                     ((Number) commandLine.getParsedOptionValue("token_exp_minutes")).intValue();
147             }
148             if (commandLine.hasOption("mqtt_bridge_hostname")) {
149                 res.mqttBridgeHostname = commandLine.getOptionValue("mqtt_bridge_hostname");
150             }
151             if (commandLine.hasOption("mqtt_bridge_port")) {
152                 res.mqttBridgePort =
153                     ((Number) commandLine.getParsedOptionValue("mqtt_bridge_port")).shortValue();
154             }
155             if (commandLine.hasOption("message_type")) {
156                 res.messageType = commandLine.getOptionValue("message_type");
157             }
158             return res;
159         } catch (ParseException e) {
160             System.err.println(e.getMessage());
161             return null;
162         }
163     }
164 }
