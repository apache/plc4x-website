19  package org.apache.plc4x.java.ads.model;
20  
21  import org.apache.plc4x.java.ads.api.util.ByteValue;
22  import org.apache.plc4x.java.api.exceptions.PlcInvalidFieldException;
23  
24  import java.util.Objects;
25  import java.util.regex.Matcher;
26  import java.util.regex.Pattern;
27  
28  /**
29   * ADS address witch is defined by {@code indexGroup/indexOffset}. These values can be either supplied as int or hex
30   * representation.
31   */
32  public class DirectAdsField implements AdsField {
33  
34      private static final Pattern RESOURCE_ADDRESS_PATTERN = Pattern.compile("^((0[xX](?<indexGroupHex>[0-9a-fA-F]+))|(?<indexGroup>\\d+))/((0[xX](?<indexOffsetHex>[0-9a-fA-F]+))|(?<indexOffset>\\d+)):(?<adsDataType>\\w+)(\\[(?<numberOfElements>\\d)])?");
35  
36      private final long indexGroup;
37  
38      private final long indexOffset;
39  
40      private final AdsDataType adsDataType;
41  
42      private final int numberOfElements;
43  
44      private DirectAdsField(long indexGroup, long indexOffset, AdsDataType adsDataType, Integer numberOfElements) {
45          ByteValue.checkUnsignedBounds(indexGroup, 4);
46          this.indexGroup = indexGroup;
47          ByteValue.checkUnsignedBounds(indexOffset, 4);
48          this.indexOffset = indexOffset;
49          this.adsDataType = Objects.requireNonNull(adsDataType);
50          this.numberOfElements = numberOfElements != null ? numberOfElements : 1;
51          if (this.numberOfElements <= 0) {
52              throw new IllegalArgumentException("numberOfElements must be greater then zero. Was " + this.numberOfElements);
53          }
54      }
55  
56      public static DirectAdsField of(long indexGroup, long indexOffset, AdsDataType adsDataType, Integer numberOfElements) {
57          return new DirectAdsField(indexGroup, indexOffset, adsDataType, numberOfElements);
58      }
59  
60      public static DirectAdsField of(String address) {
61          Matcher matcher = RESOURCE_ADDRESS_PATTERN.matcher(address);
62          if (!matcher.matches()) {
63              throw new PlcInvalidFieldException(address, RESOURCE_ADDRESS_PATTERN, "{indexGroup}/{indexOffset}:{adsDataType}([numberOfElements])?");
64          }
65  
66          String indexGroupStringHex = matcher.group("indexGroupHex");
67          String indexGroupString = matcher.group("indexGroup");
68  
69          String indexOffsetStringHex = matcher.group("indexOffsetHex");
70          String indexOffsetString = matcher.group("indexOffset");
71  
72          long indexGroup;
73          if (indexGroupStringHex != null) {
74              indexGroup = Long.parseLong(indexGroupStringHex, 16);
75          } else {
76              indexGroup = Long.parseLong(indexGroupString);
77          }
78  
79          long indexOffset;
80          if (indexOffsetStringHex != null) {
81              indexOffset = Long.parseLong(indexOffsetStringHex, 16);
82          } else {
83              indexOffset = Long.parseLong(indexOffsetString);
84          }
85  
86          String adsDataTypeString = matcher.group("adsDataType");
87          AdsDataType adsDataType = AdsDataType.valueOf(adsDataTypeString);
88  
89          String numberOfElementsString = matcher.group("numberOfElements");
90          Integer numberOfElements = numberOfElementsString != null ? Integer.valueOf(numberOfElementsString) : null;
91  
92          return new DirectAdsField(indexGroup, indexOffset, adsDataType, numberOfElements);
93      }
94  
95      public static boolean matches(String address) {
96          return RESOURCE_ADDRESS_PATTERN.matcher(address).matches();
97      }
98  
99      public long getIndexGroup() {
100         return indexGroup;
101     }
102 
103     public long getIndexOffset() {
104         return indexOffset;
105     }
106 
107     @Override
108     public AdsDataType getAdsDataType() {
109         return adsDataType;
110     }
111 
112     public int getNumberOfElements() {
113         return numberOfElements;
114     }
115 
116     @Override
117     public boolean equals(Object o) {
118         if (this == o) {
119             return true;
120         }
121         if (!(o instanceof DirectAdsField)) {
122             return false;
123         }
124         DirectAdsField that = (DirectAdsField) o;
125         return indexGroup == that.indexGroup &&
126             indexOffset == that.indexOffset;
127     }
128 
129     @Override
130     public int hashCode() {
131         return Objects.hash(indexGroup, indexOffset);
132     }
133 
134     @Override
135     public String toString() {
136         return "DirectAdsField{" +
137             "indexGroup=" + indexGroup +
138             ", indexOffset=" + indexOffset +
139             '}';
140     }
141 }
