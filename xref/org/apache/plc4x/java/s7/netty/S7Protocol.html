60  /**
61   * Communication Layer between the Application level ({@link Plc4XS7Protocol}) and the lower level (tcp) that sends and receives {@link S7Message}s.
62   * This layer also handles the control over the "wire", i.e., the queues of incoming and outgoing messages.
63   * Furthermore, here {@link S7Message}s are marshalled and unmarshalled to {@link ByteBuf}s to be send over wire.
64   *
65   * Before messages are send to the wire an optional {@link S7MessageProcessor} can be applied.
66   *
67   * @see S7MessageProcessor
68   */

112     @Override

114         this.queue = new PendingWriteQueue(ctx);
115         try {
116             Field prevField = FieldUtils.getField(ctx.getClass(), "prev", true);
117             if(prevField != null) {
118                 ChannelHandlerContext prevContext = (ChannelHandlerContext) prevField.get(ctx);
119                 prevChannelHandler = prevContext.handler();
120             }
121         } catch(Exception e) {
122             logger.error("Error accessing field 'prev'", e);
123         }
124     }

128         this.queue.removeAndWriteAll();

134         // Send everything so we get a proper failure for those pending writes
135         this.queue.removeAndWriteAll();
136         super.channelInactive(ctx);
137     }

140      * If the S7 protocol layer is used over Iso TP, then after receiving a {@link IsoTPConnectedEvent} the
141      * corresponding S7 setup communication message has to be sent in order to negotiate the S7 protocol layer.
142      *
143      * @param ctx the current protocol layers context
144      * @param evt the event
145      * @throws Exception throws an exception if something goes wrong internally
146      */
147     @Override
148     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
149         // If we are using S7 inside of IsoTP, then we need to intercept IsoTPs connected events.
150         if ((prevChannelHandler instanceof IsoTPProtocol) && (evt instanceof IsoTPConnectedEvent)) {
151             // Setup Communication
152             SetupCommunicationRequestMessage setupCommunicationRequest =
153                 new SetupCommunicationRequestMessage((short) 0, maxAmqCaller, maxAmqCallee, pduSize, null);
154 
155             ctx.channel().writeAndFlush(setupCommunicationRequest);
156         }
157 
158         else {
159             super.userEventTriggered(ctx, evt);
160         }
161     }

204     private void writeS7Message(Channel channel, PromiseCombiner promiseCombiner,
205                                 S7Message message, ByteBuf buf) throws PlcProtocolException {
206         encodeHeader(message, buf);
207         encodeParameters(message, buf);
208         encodePayloads(message, buf);
209 
210         // Check if the message doesn't exceed the negotiated maximum size.
211         if (buf.writerIndex() > pduSize) {
212             throw new PlcProtocolPayloadTooBigException("s7", pduSize, buf.writerIndex(), message);
213         } else {
214             ChannelPromise subPromise = new DefaultChannelPromise(channel);
215             // The tpduRef was 0x01 but had to be changed to 0x00 in order to support Siemens LOGO devices.
216             queue.add(new DataTpdu(true, (byte) 0x00, Collections.emptyList(), buf, message), subPromise);
217             promiseCombiner.add((Future) subPromise);
218             logger.debug("S7 Message with id {}Â queued", message.getTpduReference());
219         }
220     }

222     private void encodePayloads(S7Message in, ByteBuf buf) throws PlcProtocolException {
223         if(in.getPayloads() != null) {
224             Iterator<S7Payload> payloadIterator = in.getPayloads().iterator();
225             while(payloadIterator.hasNext()) {
226                 S7Payload payload = payloadIterator.next();
227                 switch (payload.getType()) {
228                     case WRITE_VAR:
229                         encodeWriteVarPayload((VarPayload) payload, buf, !payloadIterator.hasNext());
230                         break;
231                     case CPU_SERVICES:
232                         encodeCpuServicesPayload((CpuServicesPayload) payload, buf);
233                         break;
234                     default:
235                         throw new PlcProtocolException("Writing payloads of type " +
236                             payload.getType().name() + " not implemented.");
237                 }
238             }
239         }
240     }

242     private void encodeWriteVarPayload(VarPayload varPayload, ByteBuf buf, boolean lastItem) {
243         for (VarPayloadItem payloadItem : varPayload.getItems()) {
244             buf.writeByte(payloadItem.getReturnCode().getCode());
245             buf.writeByte(payloadItem.getDataTransportSize().getCode());
246             // TODO: Check if this is correct?!?! Might be problems with sizeInBits = true/false
247             buf.writeShort(payloadItem.getData().length);
248             buf.writeBytes(payloadItem.getData());
249             // if this is not the last item and it's payload is exactly one byte, we need to output a fill-byte.
250             if((payloadItem.getData().length == 1) && !lastItem) {
251                 buf.writeByte(0x00);
252             }
253         }
254     }

301     private void encodeParameters(S7Message in, ByteBuf buf) throws PlcProtocolException {
302         for (S7Parameter s7Parameter : in.getParameters()) {
303             buf.writeByte(s7Parameter.getType().getCode());
304             switch (s7Parameter.getType()) {
305                 case READ_VAR:
306                 case WRITE_VAR:
307                     encodeParameterReadWriteVar(buf, (VarParameter) s7Parameter);
308                     break;
309                 case SETUP_COMMUNICATION:
310                     encodeParameterSetupCommunication(buf, (SetupCommunicationParameter) s7Parameter);
311                     break;
312                 case CPU_SERVICES:
313                     encodeCpuServicesParameter(buf, (CpuServicesParameter) s7Parameter);
314                     break;
315                 default:
316                     throw new PlcProtocolException("Writing parameters of type " +
317                         s7Parameter.getType().name() + " not implemented.");
318             }
319         }
320     }

322     private void encodeHeader(S7Message in, ByteBuf buf) {
323         buf.writeByte(S7_PROTOCOL_MAGIC_NUMBER);
324         buf.writeByte(in.getMessageType().getCode());
325         // Reserved (is always constant 0x0000)
326         buf.writeShort((short) 0x0000);
327         // PDU Reference (Request Id, generated by the initiating node)
328         buf.writeShort(in.getTpduReference());
329         // S7 message parameters length
330         buf.writeShort(S7SizeHelper.getParametersLength(in.getParameters()));
331         // Data field length
332         buf.writeShort(S7SizeHelper.getPayloadsLength(in.getPayloads()));

349     private void encodeParameterReadWriteVar(ByteBuf buf, VarParameter s7Parameter) throws PlcProtocolException {
350         List<VarParameterItem> items = s7Parameter.getItems();
351         // PlcReadRequestItem count (Read one variable at a time)
352         buf.writeByte((byte) items.size());
353         for (VarParameterItem item : items) {
354             VariableAddressingMode addressMode = item.getAddressingMode();
355             if (addressMode == VariableAddressingMode.S7ANY) {
356                 encodeS7AnyParameterItem(buf, (S7AnyVarParameterItem) item);
357             } else {
358                 throw new PlcProtocolException("Writing VarParameterItems with addressing mode " +
359                     addressMode.name() + " not implemented");
360             }
361         }
362     }

389     private void encodeS7AnyParameterItem(ByteBuf buf, S7AnyVarParameterItem s7AnyRequestItem) {
390         buf.writeByte(s7AnyRequestItem.getSpecificationType().getCode());
391         // Length of this item (excluding spec type and length)
392         buf.writeByte((byte) 0x0a);
393         buf.writeByte(s7AnyRequestItem.getAddressingMode().getCode());
394         buf.writeByte(s7AnyRequestItem.getDataType().getTypeCode());
395         buf.writeShort(s7AnyRequestItem.getNumElements());
396         buf.writeShort(s7AnyRequestItem.getDataBlockNumber());
397         buf.writeByte(s7AnyRequestItem.getMemoryArea().getCode());
398         // A S7 address is 3 bytes long. Unfortunately the byte-offset is NOT located in
399         // byte 1 and byte 2 and the bit offset in byte 3. Siemens used the last 3 bits of
400         // byte 3 for the bit-offset and the remaining 5 bits of byte 3 to contain the lowest
401         // 5 bits of the byte-offset. The highest 5 bits of byte 1 are probably left unused
402         // for future extensions.
403         buf.writeShort((short) (s7AnyRequestItem.getByteOffset() >> 5));
404         buf.writeByte((byte) ((
405                 (s7AnyRequestItem.getByteOffset() & 0x1F) << 3)
406                 | (s7AnyRequestItem.getBitOffset() & 0x07)));
407     }

419     protected void decode(ChannelHandlerContext ctx, IsoTPMessage in, List<Object> out) {
420         if (logger.isTraceEnabled()) {
421             logger.trace("Got Data: {}", ByteBufUtil.hexDump(in.getUserData()));
422         }
423         ByteBuf userData = in.getUserData();
424         if (userData.readableBytes() == 0) {
425             return;
426         }
427 
428         if (userData.readByte() != S7_PROTOCOL_MAGIC_NUMBER) {
429             logger.warn("Expecting S7 protocol magic number.");
430             if (logger.isDebugEnabled()) {
431                 logger.debug("Got Data: {}", ByteBufUtil.hexDump(userData));
432             }
433             return;
434         }
435 
436         MessageType messageType = MessageType.valueOf(userData.readByte());
437         boolean isResponse = messageType == MessageType.ACK_DATA;
438         userData.readShort();  // Reserved (is always constant 0x0000)
439         short tpduReference = userData.readShort();
440         short headerParametersLength = userData.readShort();
441         short userDataLength = userData.readShort();
442         byte errorClass = 0;
443         byte errorCode = 0;
444         if (isResponse) {
445             errorClass = userData.readByte();
446             errorCode = userData.readByte();
447         }
448 
449         List<S7Parameter> s7Parameters = new LinkedList<>();
450         int i = 0;
451 
452         while (i < headerParametersLength) {
453             S7Parameter parameter = decodeParameter(userData, isResponse);
454             s7Parameters.add(parameter);
455             if (parameter instanceof SetupCommunicationParameter) {
456                 handleSetupCommunications(ctx, (SetupCommunicationParameter) parameter);
457             }
458             i += S7SizeHelper.getParameterLength(parameter);
459         }
460 
461         List<S7Payload> s7Payloads = decodePayloads(userData, isResponse, userDataLength, s7Parameters);
462 
463         logger.debug("S7 Message with id {}Â received", tpduReference);
464 
465         if (isResponse) {
466             S7ResponseMessageges/S7ResponseMessage.html#S7ResponseMessage">S7ResponseMessage responseMessage = new S7ResponseMessage(
467                 messageType, tpduReference, s7Parameters, s7Payloads, errorClass, errorCode);
468 
469             // Remove the current response from the list of unconfirmed messages.
470             DataTpdu requestTpdu = sentButUnacknowledgedTpdus.remove(tpduReference);
471 
472             // Get the corresponding request message.
473             S7RequestMessagee.html#S7RequestMessage">S7RequestMessage requestMessage = (requestTpdu != null) ? (S7RequestMessage) requestTpdu.getParent() : null;
474 
475             if(requestMessage != null) {
476                 // Set this individual request to "acknowledged".
477                 requestMessage.setAcknowledged(true);
478 
479                 // Give the request and response to a message processor to process the incoming message.
480                 if(messageProcessor != null) {
481                     try {
482                         responseMessage = messageProcessor.processResponse(requestMessage, responseMessage);
483                     } catch(Exception e) {
484                         logger.error("Error processing message", e);
485                         ctx.fireExceptionCaught(e);
486                         return;
487                     }
488                 }
489 
490                 if(responseMessage != null) {
491                     out.add(responseMessage);
492 
493                     // If this is a USER_DATA packet the probability is high that this is
494                     // a response to the identification request, we have to handle that.
495                     if(responseMessage.getMessageType() == MessageType.USER_DATA) {
496                         for (S7Payload payload : responseMessage.getPayloads()) {
497                             if(payload instanceof CpuServicesPayload) {
498                                 handleIdentifyRemote(ctx, (CpuServicesPayload) payload);
499                             }
500                         }
501                     }
502                 }
503 
504                 // Try to send the next message (if there is one).
505                 trySendingMessages(ctx);
506             }
507 
508         } else {
509             // CpuService responses are encoded as requests.
510             for (S7Parameter s7Parameter : s7Parameters) {
511                 // Only if we have a response parameter, the payload is a response payload.
512                 if(s7Parameter instanceof CpuServicesResponseParameter) {
513                     for (S7Payload s7Payload : s7Payloads) {
514                         if(s7Payload instanceof CpuServicesPayload) {
515                             CpuServicesPayloadache/plc4x/java/s7/netty/model/payloads/CpuServicesPayload.html#CpuServicesPayload">CpuServicesPayload cpuServicesPayload = (CpuServicesPayload) s7Payload;
516                             handleIdentifyRemote(ctx, cpuServicesPayload);
517                         }
518                     }
519                 }
520             }
521             out.add(new S7RequestMessage(messageType, tpduReference, s7Parameters, s7Payloads, null));
522         }
523     }

525     private void handleSetupCommunications(ChannelHandlerContext ctx, SetupCommunicationParameter setupCommunicationParameter) {
526         maxAmqCaller = setupCommunicationParameter.getMaxAmqCaller();
527         maxAmqCallee = setupCommunicationParameter.getMaxAmqCallee();
528         pduSize = setupCommunicationParameter.getPduLength();
529 
530         logger.info("S7Connection established pdu-size {}, max-amq-caller {}, " +
531                 "max-amq-callee {}", pduSize, maxAmqCaller, maxAmqCallee);
532 
533         // Only if the controller type is set to "ANY", then try to identify the PLC type.
534         if(controllerType == S7ControllerType.ANY) {
535             // Prepare a message to request the remote to identify itself.
536             S7RequestMessage7RequestMessage.html#S7RequestMessage">S7RequestMessage identifyRemoteMessage = new S7RequestMessage(MessageType.USER_DATA, (short) 2,
537                 Collections.singletonList(new CpuServicesRequestParameter(
538                     CpuServicesParameterFunctionGroup.CPU_FUNCTIONS,
539                     CpuServicesParameterSubFunctionGroup.READ_SSL, (byte) 0)),
540                 Collections.singletonList(new CpuServicesPayload(DataTransportErrorCode.OK, SslId.MODULE_IDENTIFICATION,
541                     (short) 0x0000)), null);
542             ctx.channel().writeAndFlush(identifyRemoteMessage);
543         }
544         // If a concrete type was specified, then we're done here.
545         else {
546             if(logger.isInfoEnabled()) {
547                 logger.info(String.format("Successfully connected to S7: %s", controllerType.name()));
548                 logger.info(String.format("- max amq caller: %s", maxAmqCaller));
549                 logger.info(String.format("- max amq callee: %s", maxAmqCallee));
550                 logger.info(String.format("- pdu size: %s", pduSize));
551             }
552 
553             // Send an event that connection setup is complete.
554             ctx.channel().pipeline().fireUserEventTriggered(new S7ConnectedEvent());
555         }

633     private CpuServicesPayload decodeCpuServicesPayload(ByteBuf userData) {
634         DataTransportErrorCode returnCode = DataTransportErrorCode.valueOf(userData.readByte());
635         DataTransportSize dataTransportSize = DataTransportSize.valueOf(userData.readByte());
636         if(dataTransportSize != DataTransportSize.OCTET_STRING) {
637             // TODO: Output an error.
638         }
639         short length = userData.readShort();
640         SslId sslId = SslId.valueOf(userData.readShort());
641         short sslIndex = userData.readShort();
642         // If the length is 4 there is no `partial list length in bytes` and `partial list count` parameters.
643         if(length == 4) {
644             return new CpuServicesPayload(returnCode, sslId, sslIndex);
645         }
646         // If the length is not 4, then it has to be at least 8.
647         else if(length >= 8) {
648             // TODO: We should probably ensure we don't read more than this.
649             // Skip the partial list length in words.
650             userData.skipBytes(2);
651             short partialListCount = userData.readShort();
652             List<SslDataRecord> sslDataRecords = new LinkedList<>();
653             for(int i = 0; i < partialListCount; i++) {
654                 short index = userData.readShort();
655                 byte[] articleNumberBytes = new byte[20];
656                 userData.readBytes(articleNumberBytes);
657                 String articleNumber = new String(articleNumberBytes, StandardCharsets.UTF_8).trim();
658                 short bgType = userData.readShort();
659                 short moduleOrOsVersion = userData.readShort();
660                 short pgDescriptionFileVersion = userData.readShort();
661                 sslDataRecords.add(new SslModuleIdentificationDataRecord(
662                     index, articleNumber, bgType, moduleOrOsVersion, pgDescriptionFileVersion));
663             }
664             return new CpuServicesPayload(returnCode, sslId, sslIndex, sslDataRecords);
665         }
666         // In all other cases, it's probably an error.
667         else {
668             // TODO: Output an error.
669         }
670         return null;
671     }

673     private S7Parameter decodeParameter(ByteBuf in, boolean isResponse) {
674         ParameterType parameterType = ParameterType.valueOf(in.readByte());
675         if (parameterType == null) {
676             logger.error("Could not find parameter type");
677             return null;
678         }
679         switch (parameterType) {
680             case CPU_SERVICES:
681                 return decodeCpuServicesParameter(in);
682             case READ_VAR:
683             case WRITE_VAR:
684                 List<VarParameterItem> varParameterItems;
685                 byte numItems = in.readByte();
686                 if (!isResponse) {
687                     varParameterItems = decodeReadWriteVarParameter(in, numItems);
688                 } else {
689                     varParameterItems = Collections.singletonList(
690                         new S7AnyVarParameterItem(null, null, null, numItems, (short) 0, (short) 0, (byte) 0));
691                 }
692                 return new VarParameter(parameterType, varParameterItems);
693             case SETUP_COMMUNICATION:
694                 // Reserved (is always constant 0x00)
695                 in.readByte();
696                 short callingMaxAmq = in.readShort();
697                 short calledMaxAmq = in.readShort();
698                 short pduLength = in.readShort();
699                 return new SetupCommunicationParameter(callingMaxAmq, calledMaxAmq, pduLength);
700             default:
701                 if (logger.isErrorEnabled()) {
702                     logger.error("Unimplemented parameter type: {}", parameterType.name());
703                 }
704         }
705         return null;
706     }

708     private CpuServicesParameter decodeCpuServicesParameter(ByteBuf in) {
709         if(in.readShort() != 0x0112) {
710             if (logger.isErrorEnabled()) {
711                 logger.error("Expecting 0x0112 for CPU_SERVICES parameter");
712             }
713             return null;
714         }
715         byte parameterLength = in.readByte();
716         if((parameterLength != 4) && (parameterLength != 8)) {
717             if (logger.isErrorEnabled()) {
718                 logger.error("Parameter length should be 4 or 8, but was {}", parameterLength);
719             }
720             return null;
721         }
722         // Skipping this as it sort of contains redundant information.
723         in.readByte();
724         byte typeAndFunctionGroup = in.readByte();
725         // If bit 7 is set, it's a request (if bit 8 is set it's a response).
726         boolean requestParameter = (typeAndFunctionGroup & 0x64) != 0;
727         // The last 4 bits contain the function group value.
728         typeAndFunctionGroup = (byte) (typeAndFunctionGroup & 0xF);
729         CpuServicesParameterFunctionGroup functionGroup =
730             CpuServicesParameterFunctionGroup.valueOf(typeAndFunctionGroup);
731         CpuServicesParameterSubFunctionGroup subFunctionGroup =
732             CpuServicesParameterSubFunctionGroup.valueOf(in.readByte());
733         byte sequenceNumber = in.readByte();
734         if(!requestParameter) {
735             return new CpuServicesRequestParameter(functionGroup, subFunctionGroup, sequenceNumber);
736         } else {
737             byte dataUnitReferenceNumber = in.readByte();
738             boolean lastDataUnit = in.readByte() == 0x00;
739             ParameterError error = ParameterError.valueOf(in.readShort());
740             return new CpuServicesResponseParameter(functionGroup, subFunctionGroup, sequenceNumber,
741                 dataUnitReferenceNumber, lastDataUnit, error);
742         }
743     }

746         List<VarParameterItem> items = new LinkedList<>();
747         for (int i = 0; i < numItems; i++) {
748             SpecificationType specificationType = SpecificationType.valueOf(in.readByte());
749             // Length of the rest of this item.
750             byte itemLength = in.readByte();
751             if (itemLength != 0x0a) {
752                 logger.warn("Expecting a length of 10 here.");
753                 return items;
754             }
755             VariableAddressingMode variableAddressingMode = VariableAddressingMode.valueOf(in.readByte());
756             if (variableAddressingMode == VariableAddressingMode.S7ANY) {
757                 TransportSize dataType = TransportSize.valueOf(in.readByte());
758                 short length = in.readShort();
759                 short dbNumber = in.readShort();
760                 MemoryArea memoryArea = MemoryArea.valueOf(in.readByte());
761                 short byteAddress = (short) (in.readShort() << 5);
762                 byte tmp = in.readByte();
763                 // Only the least 3 bits are the bit address, the
764                 byte bitAddress = (byte) (tmp & 0x07);

820     private S7ControllerType lookupControllerType(String articleNumber) {
821         if(!articleNumber.startsWith("6ES7 ")) {
822             return S7ControllerType.ANY;
823         }
824 
825         String model = articleNumber.substring(articleNumber.indexOf(' ') + 1, articleNumber.indexOf(' ') + 2);
826         switch (model) {
827             case "2":
828                 return S7ControllerType.S7_1200;
829             case "5":
830                 return S7ControllerType.S7_1500;
831             case "3":
832                 return S7ControllerType.S7_300;
833             case "4":
834                 return S7ControllerType.S7_400;
835             default:
836                 if(logger.isInfoEnabled()) {
837                     logger.info(String.format("Looking up unknown article number %s", articleNumber));
838                 }
839                 return S7ControllerType.ANY;
