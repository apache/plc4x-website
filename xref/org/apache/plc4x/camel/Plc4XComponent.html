19  package org.apache.plc4x.camel;
20  
21  import org.apache.camel.Endpoint;
22  import org.apache.camel.impl.DefaultComponent;
23  
24  import java.util.Map;
25  
26  public class Plc4XComponent extends DefaultComponent {
27  
28      @Override
29      protected Endpoint createEndpoint(String uri, String remaining, Map<String, Object> parameters) throws Exception {
30          Endpoint endpoint = new Plc4XEndpoint(uri, this);
31          setProperties(endpoint, parameters);
32          return endpoint;
33      }
34  
35      @Override
36      protected void afterConfiguration(String uri, String remaining, Endpoint endpoint, Map<String, Object> parameters) {
37          Plc4XEndpoint plc4XEndpoint = (Plc4XEndpoint) endpoint;
38          plc4XEndpoint.setDriver(remaining.split(":")[0]);
39      }
40  }
