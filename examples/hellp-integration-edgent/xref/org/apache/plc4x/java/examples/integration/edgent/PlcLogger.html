19  package org.apache.plc4x.java.examples.integration.edgent;
20  
21  import org.apache.edgent.function.Supplier;
22  import org.apache.edgent.providers.direct.DirectProvider;
23  import org.apache.edgent.topology.TStream;
24  import org.apache.edgent.topology.Topology;
25  import org.apache.plc4x.edgent.PlcConnectionAdapter;
26  import org.apache.plc4x.edgent.PlcFunctions;
27  
28  import java.util.concurrent.TimeUnit;
29  
30  public class PlcLogger {
31  
32      public static void main(String[] args) throws Exception {
33          CliOptions options = CliOptions.fromArgs(args);
34          if (options == null) {
35              CliOptions.printHelp();
36              // Could not parse.
37              System.exit(1);
38          }
39  
40          // Get a plc connection.
41          try (PlcConnectionAdapter plcAdapter = new PlcConnectionAdapter(options.getConnectionString())) {
42              // Initialize the Edgent core.
43              DirectProvider dp = new DirectProvider();
44              Topology top = dp.newTopology();
45  
46              // Define the event stream.
47              // 1) PLC4X source generating a stream of bytes.
48              Supplier<Byte> plcSupplier = PlcFunctions.byteSupplier(plcAdapter, options.getFieldAddress());
49              // 2) Use polling to get an item from the byte-stream in regular intervals.
50              TStream<Byte> source = top.poll(plcSupplier, options.getPollingInterval(), TimeUnit.MILLISECONDS);
51              // 3) Output the events in the stream on the console.
52              source.print();
53  
54              // Submit the topology and hereby start the event streams.
55              dp.submit(top);
56          }
57      }
58  
59  }
