43  /**
44   * PlcConnectionAdapter encapsulates a plc4x {@link PlcConnection}.
45   * <p>
46   * The idea here is to use PlcConnectionAdapter to enable our Edgent Supplier/Consumer
47   * instances to be isolated from the details of / variability of
48   * PlcConnection mgmt and such.
49   * <p>
50   * A PlcConnectionAdapter is subject to the constraints of the underlying
51   * PlcConnection for the device.
52   * <p>
53   * A single PlcConnectionAdaptor represents a single underlying PlcConnection/connection
54   * to a plc device.
55   * <p>
56   * Multiple PlcConnectionAdaptor instances, hence multiple PlcConnection instances,
57   * can be created for a single plc device subject to the underlying device's protocol's
58   * support for multiple connections from a single client.
59   * <p>
60   * A single PlcConnectionAdapter can be used by multiple threads concurrently
61   * (e.g., used by multiple PlcFunctions Consumers for {@code Topology.poll()} and/or
62   * multiple Suppliers for {@code TStream.sink()}).
63   *
64   * @see PlcFunctions
65   */
66  public class PlcConnectionAdapter implements AutoCloseable {
67  
68      private static final Logger logger = LoggerFactory.getLogger(PlcConnectionAdapter.class);
69  
70      private static final String FIELD_NAME = "default";
71  
72      private String plcConnectionUrl;
73      private PlcConnection plcConnection;
74  
75      /*
76       * NOTES:
77       * - if we get to the point of the application needing some feedback (possibly control)
78       *   of read or write errors, my thinking is to enhance the PlcConnectionAdapter
79       *   to enable the app to register an error callback handler or such.
80       */
81  
82      public PlcConnectionAdapter(PlcConnection plcConnection) {
83          this.plcConnection = plcConnection;
84      }
85  
86      public PlcConnectionAdapter(String plcConnectionUrl) {
87          this.plcConnectionUrl = plcConnectionUrl;
88      }
89  
90      PlcConnection getConnection() throws PlcException {
91          synchronized (this) {
92              if (plcConnection == null) {
93                  plcConnection = new PlcDriverManager().getConnection(plcConnectionUrl);
94              }
95              return plcConnection;
96          }
97      }
98  
99      @Override
100     public void close() throws Exception {
101         // only close a connection this instance created/connected
102         if (plcConnectionUrl != null && plcConnection != null) {
103             plcConnection.close();
104         }
105     }
106 
107     public PlcReadRequest.Builder readRequestBuilder() throws PlcException {
108         return getConnection().readRequestBuilder();
109     }

111     Supplier<PlcReadResponse> newSupplier(PlcReadRequest readRequest) {
112         return new Supplier<PlcReadResponse>() {
113             private static final long serialVersionUID = 1L;
114 
115             @Override
116             public PlcReadResponse get() {
117                 PlcConnection connection = null;
118                 try {
119                     connection = getConnection();
120                     LocalDateTime start = LocalDateTime.now();
121                     PlcReadResponse plcReadResponse = readRequest.execute().get();
122                     LocalDateTime end = LocalDateTime.now();
123                     if(logger.isTraceEnabled()) {
124                         long diff = ChronoUnit.MILLIS.between(start, end);
125                         logger.trace("Processed request in " + diff + "ms");
126                     }
127                     return plcReadResponse;
128                 } catch (Exception e) {
129                     logger.error("reading from plc device {} {} failed", connection, readRequest, e);
130                     return null;
131                 }
132             }
133         };
134     }

146     private class MySupplier<T> implements Supplier<T> {
147 
148         private static final long serialVersionUID = 1L;
149 
150         private Class<T> genericDatatype;
151         private PlcClientDatatype clientDatatype;
152         private String fieldQuery;
153 
154         MySupplier(Class<T> genericDatatype, PlcClientDatatype clientDatatype, String fieldQuery) {
155             this.genericDatatype = genericDatatype;
156             this.clientDatatype = clientDatatype;
157             this.fieldQuery = fieldQuery;
158         }
159 
160         @Override
161         @SuppressWarnings("unchecked")
162         public T get() {
163             PlcConnection connection = null;
164             PlcField field = null;
165             try {
166                 connection = getConnection();
167                 PlcReadRequest readRequest = connection.readRequestBuilder().addItem(FIELD_NAME, fieldQuery).build();
168                 PlcReadResponse readResponse = readRequest.execute().get();
169                 Object value = null;
170                 switch (clientDatatype) {
171                     case BOOLEAN:
172                         value = readResponse.getBoolean(FIELD_NAME);
173                         break;
174                     case BYTE:
175                         value = readResponse.getByte(FIELD_NAME);
176                         break;
177                     case SHORT:
178                         value = readResponse.getShort(FIELD_NAME);
179                         break;
180                     case INTEGER:
181                         value = readResponse.getInteger(FIELD_NAME);
182                         break;
183                     case LONG:
184                         value = readResponse.getLong(FIELD_NAME);
185                         break;
186                     case FLOAT:
187                         value = readResponse.getFloat(FIELD_NAME);
188                         break;
189                     case DOUBLE:
190                         value = readResponse.getDouble(FIELD_NAME);
191                         break;
192                     case STRING:
193                         value = readResponse.getString(FIELD_NAME);
194                         break;
195                     case TIME:
196                         value = readResponse.getTime(FIELD_NAME);
197                         break;
198                     case DATE:
199                         value = readResponse.getDate(FIELD_NAME);
200                         break;
201                     case DATE_TIME:
202                         value = readResponse.getDateTime(FIELD_NAME);
203                         break;
204                 }
205                 if (value != null) {
206                     if (genericDatatype.isAssignableFrom(value.getClass())) {
207                         return (T) value;
208                     } else {
209                         logger.error("types don't match {} should be of type {}", value.getClass(), genericDatatype);
210                     }
211                 }
212             } catch (Exception e) {
213                 logger.error("reading from plc device {} {} failed", connection, field, e);
214             }
215             return null;
216         }
217     }

249                         break;
250                     case SHORT:
251                         value = readResponse.getAllShorts(FIELD_NAME);
252                         break;
253                     case INTEGER:
254                         value = readResponse.getAllIntegers(FIELD_NAME);
255                         break;
256                     case LONG:
257                         value = readResponse.getAllLongs(FIELD_NAME);
258                         break;
259                     case FLOAT:
260                         value = readResponse.getAllFloats(FIELD_NAME);
261                         break;
262                     case DOUBLE:
263                         value = readResponse.getAllDoubles(FIELD_NAME);
264                         break;
265                     case STRING:
266                         value = readResponse.getAllStrings(FIELD_NAME);
267                         break;
268                     case TIME:
269                         value = readResponse.getAllTimes(FIELD_NAME);
270                         break;
271                     case DATE:
272                         value = readResponse.getAllDates(FIELD_NAME);
273                         break;
274                     case DATE_TIME:
275                         value = readResponse.getAllDateTimes(FIELD_NAME);
276                         break;
277                 }
278                 if (value != null) {
279                     return Collections.checkedList((List<T>) value, genericDatatype);
280                 }
281             } catch (Exception e) {
282                 logger.error("reading from plc device {} {} failed", connection, field, e);
283             }
284             return null;
285         }

296     private abstract class BaseConsumer<T> implements Consumer<T> {
297 
298         protected void write(PlcClientDatatype clientDatatype, String fieldQuery, Object fieldValue) {
299             PlcConnection connection = null;
300             try {
301                 connection = getConnection();
302                 PlcWriteRequest.Builder builder = connection.writeRequestBuilder();
303                 PlcWriteRequest writeRequest = builder.build();
304                 addItem(builder, clientDatatype, fieldQuery, fieldValue);
305                 writeRequest.execute().get();
306             } catch (Exception e) {
307                 logger.error("writing to plc device {} {} failed", connection, fieldQuery, e);
308             }
309 
310         }
311 
312         private void addItem(PlcWriteRequest.Builder builder,
313                              PlcClientDatatype clientDatatype, String fieldQuery, Object fieldValue) {
314             switch (clientDatatype) {
315                 case BYTE:
316                     if (fieldValue instanceof Byte) {
317                         builder.addItem(FIELD_NAME, fieldQuery, (Byte) fieldValue);
318                     }
319                     break;
320                 case SHORT:
321                     if (fieldValue instanceof Short) {
322                         builder.addItem(FIELD_NAME, fieldQuery, (Short) fieldValue);
323                     }
324                     break;
325                 case INTEGER:
326                     if (fieldValue instanceof Integer) {
327                         builder.addItem(FIELD_NAME, fieldQuery, (Integer) fieldValue);
328                     }
329                     break;
330                 case LONG:
331                     if (fieldValue instanceof Long) {
332                         builder.addItem(FIELD_NAME, fieldQuery, (Long) fieldValue);
333                     }
334                     break;
335                 case FLOAT:
336                     if (fieldValue instanceof Float) {
337                         builder.addItem(FIELD_NAME, fieldQuery, (Float) fieldValue);
338                     }
339                     break;
340                 case DOUBLE:
341                     if (fieldValue instanceof Double) {
342                         builder.addItem(FIELD_NAME, fieldQuery, (Double) fieldValue);
343                     }
344                     break;
345                 case STRING:
346                     if (fieldValue instanceof String) {
347                         builder.addItem(FIELD_NAME, fieldQuery, (String) fieldValue);
348                     }
349                     break;
350                 case TIME:
351                     if (fieldValue instanceof LocalTime) {
352                         builder.addItem(FIELD_NAME, fieldQuery, (LocalTime) fieldValue);
353                     }
354                     break;
355                 case DATE:
356                     if (fieldValue instanceof LocalDate) {
357                         builder.addItem(FIELD_NAME, fieldQuery, (LocalDate) fieldValue);
358                     }
359                     break;
360                 case DATE_TIME:
361                     if (fieldValue instanceof LocalDateTime) {
362                         builder.addItem(FIELD_NAME, fieldQuery, (LocalDateTime) fieldValue);
363                     }
364                     break;
365             }
366         }
367     }

369     private class ObjectConsumer<T> extends BaseConsumer<T> {
370         private static final long serialVersionUID = 1L;
371 
372         private PlcClientDatatype clientDatatype;
373         private String fieldQuery;
374 
375         ObjectConsumer(Class<T> genericDatatype, PlcClientDatatype clientDatatype, String fieldQuery) {
376             this.clientDatatype = clientDatatype;
377             this.fieldQuery = fieldQuery;
378         }
379 
380         @Override
381         public void accept(Object fieldValue) {
382             write(clientDatatype, fieldQuery, fieldValue);
383         }
384     }

386     private class JsonConsumer<T> extends BaseConsumer<JsonObject> {
387         private static final long serialVersionUID = 1L;
388 
389         private PlcClientDatatype clientDatatype;
390         private Function<JsonObject, String> fieldQueryFn;
391         private Function<JsonObject, T> fieldValueFn;
392 
393         JsonConsumer(PlcClientDatatype clientDatatype, Function<JsonObject, String> fieldQueryFn, Function<JsonObject, T> fieldValueFn) {
394             this.clientDatatype = clientDatatype;
395             this.fieldQueryFn = fieldQueryFn;
396             this.fieldValueFn = fieldValueFn;
397         }
398 
399         @Override
400         public void accept(JsonObject jsonObject) {
401             String fieldQuery = fieldQueryFn.apply(jsonObject);
402             Object fieldValue = fieldValueFn.apply(jsonObject);
403             write(clientDatatype, fieldQuery, fieldValue);
404         }
