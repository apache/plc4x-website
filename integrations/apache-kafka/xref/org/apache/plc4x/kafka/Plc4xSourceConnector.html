19  package org.apache.plc4x.kafka;
20  
21  import org.apache.kafka.common.config.AbstractConfig;
22  import org.apache.kafka.common.config.ConfigDef;
23  import org.apache.kafka.connect.connector.Task;
24  import org.apache.kafka.connect.source.SourceConnector;
25  import org.apache.plc4x.kafka.util.VersionUtil;
26  
27  import java.util.*;
28  import java.util.stream.Collectors;
29  
30  public class Plc4xSourceConnector extends SourceConnector {
31      private static final String TOPIC_CONFIG = "topic";
32      private static final String TOPIC_DOC = "Kafka topic to publish to";
33  
34      private static final String QUERIES_CONFIG = "queries";
35      private static final String QUERIES_DOC = "Field queries to be sent to the PLC";
36  
37      private static final String RATE_CONFIG = "rate";
38      private static final Integer RATE_DEFAULT = 1000;
39      private static final String RATE_DOC = "Polling rate";
40  
41      private static final ConfigDef CONFIG_DEF = new ConfigDef()
42          .define(TOPIC_CONFIG, ConfigDef.Type.STRING, ConfigDef.Importance.HIGH, TOPIC_DOC)
43          .define(QUERIES_CONFIG, ConfigDef.Type.LIST, ConfigDef.Importance.HIGH, QUERIES_DOC)
44          .define(RATE_CONFIG, ConfigDef.Type.INT, RATE_DEFAULT, ConfigDef.Importance.MEDIUM, RATE_DOC);
45  
46      private String topic;
47      private List<String> queries;
48      private Integer rate;
49  
50      @Override
51      public Class<? extends Task> taskClass() {
52          return Plc4xSourceTask.class;
53      }
54  
55      @Override
56      public List<Map<String, String>> taskConfigs(int maxTasks) {
57          List<Map<String, String>> configs = new LinkedList<>();
58          Map<String, List<String>> groupedByHost = new HashMap<>();
59          queries.stream().map(query -> query.split("#", 2)).collect(Collectors.groupingBy(parts -> parts[0])).forEach((host, queries) -> {
60              groupedByHost.put(host, queries.stream().map(parts -> parts[1]).collect(Collectors.toList()));
61          });
62          if (groupedByHost.size() > maxTasks) {
63              // Not enough tasks
64              // TODO: throw exception?
65              return Collections.emptyList();
66          }
67          groupedByHost.forEach((host, qs) -> {
68              Map<String, String> taskConfig = new HashMap<>();
69              taskConfig.put(Plc4xSourceTask.TOPIC_CONFIG, topic);
70              taskConfig.put(Plc4xSourceTask.URL_CONFIG, host);
71              taskConfig.put(Plc4xSourceTask.QUERIES_CONFIG, String.join(",", qs));
72              taskConfig.put(Plc4xSourceTask.RATE_CONFIG, rate.toString());
73              configs.add(taskConfig);
74          });
75          return configs;
76      }
77  
78      @Override
79      public void start(Map<String, String> props) {
80          AbstractConfig config = new AbstractConfig(CONFIG_DEF, props);
81          topic = config.getString(TOPIC_CONFIG);
82          queries = config.getList(QUERIES_CONFIG);
83          rate = config.getInt(RATE_CONFIG);
84      }
85  
86      @Override
87      public void stop() {}
88  
89      @Override
90      public ConfigDef config() {
91          return CONFIG_DEF;
92      }
93  
94      @Override
95      public String version() {
96          return VersionUtil.getVersion();
97      }
98  
99  }
